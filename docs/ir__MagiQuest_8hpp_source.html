43 // M A G I Q U E S T
44 //
45 //==============================================================================
46 // MSB first, 8 Start bits (zero), 32 wand id bits, 16 magnitude bits, one stop bit
47 // Not all start bits must be received, since protocol is MSB first and so the LSB ends up always at the right position.
48 
49 #if !defined (DOXYGEN)
50 // MagiQuest packet is both Wand ID and magnitude of swish and flick
51 union magiquest_t {
52  uint64_t llword;
53  struct {
54  uint16_t magnitude;
55  uint32_t wand_id;
56  uint8_t StartBits; // first 8 MSB start bits are zero.
57  uint8_t HighByte; // just to pad the struct out to 64 bits so we can union with llword
58  } cmd;
59 };
60 #endif // !defined (DOXYGEN)
61 
62 #define MAGIQUEST_MAGNITUDE_BITS 16 // magiquest_t.cmd.magnitude
63 #define MAGIQUEST_WAND_ID_BITS 32 // magiquest_t.cmd.wand_id
64 #define MAGIQUEST_START_BITS 8 // magiquest_t.cmd.StartBits
65 
66 #define MAGIQUEST_PERIOD 1150 // Time for a full MagiQuest "bit" (1100 - 1200 usec)
67 
68 #define MAGIQUEST_BITS (MAGIQUEST_MAGNITUDE_BITS + MAGIQUEST_WAND_ID_BITS) // 48 Size of the command without the start bits
69 // The maximum size of a packet is the sum of all 3 expected fields * 2
70 #define MAGIQUEST_PACKET_SIZE (MAGIQUEST_MAGNITUDE_BITS + MAGIQUEST_WAND_ID_BITS + MAGIQUEST_START_BITS) // 56
71 
72 /*
73  * 0 = 25% mark & 75% space across 1 period
74  * 1150 * 0.25 = 288 usec mark
75  * 1150 - 288 = 862 usec space
76  * 1 = 50% mark & 50% space across 1 period
77  * 1150 * 0.5 = 575 usec mark
78  * 1150 - 575 = 575 usec space
79  */
80 #define MAGIQUEST_UNIT (MAGIQUEST_PERIOD / 4)
81 
82 #define MAGIQUEST_ONE_MARK (2 * MAGIQUEST_UNIT) // 576
83 #define MAGIQUEST_ONE_SPACE (2 * MAGIQUEST_UNIT) // 576
84 #define MAGIQUEST_ZERO_MARK MAGIQUEST_UNIT
85 #define MAGIQUEST_ZERO_SPACE (3 * MAGIQUEST_UNIT) // 864

89 void IRsend::sendMagiQuest(uint32_t wand_id, uint16_t magnitude) {
90 
91  // Set IR carrier frequency
92  enableIROut(38);
93 
94  // 8 start bits
95  sendPulseDistanceWidthData(
96  MAGIQUEST_ONE_MARK, MAGIQUEST_ONE_SPACE, MAGIQUEST_ZERO_MARK, MAGIQUEST_ZERO_SPACE, 0, 8, PROTOCOL_IS_MSB_FIRST);
97 
98  // Data
99  sendPulseDistanceWidthData(
100  MAGIQUEST_ONE_MARK, MAGIQUEST_ONE_SPACE, MAGIQUEST_ZERO_MARK, MAGIQUEST_ZERO_SPACE, wand_id, MAGIQUEST_WAND_ID_BITS,
101  PROTOCOL_IS_MSB_FIRST);
102  sendPulseDistanceWidthData(
103  MAGIQUEST_ONE_MARK, MAGIQUEST_ONE_SPACE, MAGIQUEST_ZERO_MARK, MAGIQUEST_ZERO_SPACE, magnitude, MAGIQUEST_MAGNITUDE_BITS,
104  PROTOCOL_IS_MSB_FIRST,
105  SEND_STOP_BIT);
106  IrReceiver.restartAfterSend();
107 }

111 /*
112  * decodes a 56 bit result, which is not really compatible with standard decoder layout
113  */
114 bool IRrecv::decodeMagiQuest() {
115  magiquest_t data; // Somewhere to build our code
116  unsigned int tOffset = 1; // Skip the gap between packets
117 
118  unsigned int tMark;
119  unsigned int tSpace;
120 
121 #if defined(DEBUG)
122  char bitstring[(MAGIQUEST_PACKET_SIZE + 1)];
123  bitstring[MAGIQUEST_PACKET_SIZE] = '\0';
124 #endif
125 
126  // Check we have the right amount of data, magnitude and ID bits and at least 2 start bits + 1 stop bit
127  if (decodedIRData.rawDataPtr->rawlen < (2 * (MAGIQUEST_BITS + 3))
128  || decodedIRData.rawDataPtr->rawlen > (2 * (MAGIQUEST_PACKET_SIZE + 1))) {
129  IR_DEBUG_PRINT(F("MagiQuest: "));
130  IR_DEBUG_PRINT(F("Data length="));
131  IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
132  IR_DEBUG_PRINTLN(F(" is not between 102 and 114"));
133  return false;
134  }
135 
136  // Read the bits in
137  data.llword = 0;
138  while (tOffset < (unsigned int) (decodedIRData.rawDataPtr->rawlen - 1)) {
139  // get one mark and space pair
140  tMark = decodedIRData.rawDataPtr->rawbuf[tOffset++];
141  tSpace = decodedIRData.rawDataPtr->rawbuf[tOffset++];
142 
143  IR_TRACE_PRINT(F("MagiQuest: mark="));
144  IR_TRACE_PRINT(tMark * MICROS_PER_TICK);
145  IR_TRACE_PRINT(F(" space="));
146  IR_TRACE_PRINTLN(tSpace * MICROS_PER_TICK);
147 
148  if (matchMark(tSpace + tMark, MAGIQUEST_PERIOD)) {
149  if (tSpace > tMark) {
150  // It's a 0
151  data.llword <<= 1;
152 #if defined(DEBUG)
153  bitstring[(tOffset / 2) - 1] = '0';
154 #endif
155  } else {
156  // It's a 1
157  data.llword = (data.llword << 1) | 1;
158 #if defined(DEBUG)
159  bitstring[(tOffset / 2) - 1] = '1';
160 #endif
161  }
162  } else {
163  IR_DEBUG_PRINTLN(F("Mark and space does not match the constant MagiQuest period"));
164  return false;
165  }
166  }
167 #if defined(DEBUG)
168  Serial.println(bitstring);
169 #endif
170 
171  // Success
172  decodedIRData.protocol = MAGIQUEST;
173  decodedIRData.numberOfBits = tOffset / 2;
174  decodedIRData.flags = IRDATA_FLAGS_EXTRA_INFO | IRDATA_FLAGS_IS_MSB_FIRST;
175  decodedIRData.extra = data.cmd.magnitude;
176  decodedIRData.decodedRawData = data.cmd.wand_id;
177 
178  return true;
179 }
