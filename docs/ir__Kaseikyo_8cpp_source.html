41 //==============================================================================
42 // PPPP AAA N N AAA SSSS OOO N N IIIII CCCC
43 // P P A A NN N A A S O O NN N I C
44 // PPPP AAAAA N N N AAAAA SSS O O N N N I C
45 // P A A N NN A A S O O N NN I C
46 // P A A N N A A SSSS OOO N N IIIII CCCC
47 //==============================================================================
48 // see: http://www.hifi-remote.com/johnsfine/DecodeIR.html#Panasonic
49 // IRP notation: {37k,432}<1,-1|1,-3>(8,-4,M:8,N:8,X:4,D:4,S:8,F:8,G:8,1,-173)+ {X=M:4:0^M:4:4^N:4:0^N:4:4}
50 // see: http://www.remotecentral.com/cgi-bin/mboard/rc-pronto/thread.cgi?26152
51 // The first two (8-bit) bytes are always 2 and 32 (These identify Panasonic within the Kaseikyo standard)
52 // The next two bytes are 4 independent 4-bit fields or Device and Subdevice
53 // The second to last byte is the function and the last byte is xor of the three bytes before it.
54 // 0_______ 1_______ 2______ 3_______ 4_______ 5
55 // 76543210 76543210 76543210 76543210 76543210 76543210
56 // 00000010 00100000 Dev____ Sub Dev Fun____ XOR( B2, B3, B4)
57 
58 // LSB first, start bit + 16 Vendor + 4 Parity(of vendor) + 4 Genre1 + 4 Genre2 + 10 Command + 2 ID + 8 Parity + stop bit
59 // We reduce it to: start bit + 16 Vendor + 16 Address + 8 Command + 8 Parity + stop bit
60 //
61 #define KASEIKYO_VENDOR_ID_BITS 16
62 #define KASEIKYO_VENDOR_ID_PARITY_BITS 4
63 #define KASEIKYO_ADDRESS_BITS 12
64 #define KASEIKYO_COMMAND_BITS 8
65 #define KASEIKYO_PARITY_BITS 8
66 #define KASEIKYO_BITS (KASEIKYO_VENDOR_ID_BITS + KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_ADDRESS_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS)
67 #define KASEIKYO_UNIT 432 // Pronto 0x70 / 0x10 - I measured 17 pulses
68 
69 #define KASEIKYO_HEADER_MARK (8 * KASEIKYO_UNIT) // 3456
70 #define KASEIKYO_HEADER_SPACE (4 * KASEIKYO_UNIT) // 1728
71 
72 #define KASEIKYO_BIT_MARK KASEIKYO_UNIT
73 #define KASEIKYO_ONE_SPACE (3 * KASEIKYO_UNIT) // 1296
74 #define KASEIKYO_ZERO_SPACE KASEIKYO_UNIT
75 
76 #define KASEIKYO_AVERAGE_DURATION 56000
77 #define KASEIKYO_REPEAT_PERIOD 130000
78 #define KASEIKYO_REPEAT_SPACE (KASEIKYO_REPEAT_PERIOD - KASEIKYO_AVERAGE_DURATION) // 74 ms
79 
80 // for old decoder
81 #define KASEIKYO_DATA_BITS 32

84 /*
85  * Send with LSB first
86  * Address is sub-device << 8 + device
87  */
88 void IRsend::sendKaseikyo(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, uint16_t aVendorCode) {
89  // Set IR carrier frequency
90  enableIROut(37); // 36.7kHz is the correct frequency
91 
92  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
93  while (tNumberOfCommands > 0) {
94 
95  // Header
96  mark(KASEIKYO_HEADER_MARK);
97  space(KASEIKYO_HEADER_SPACE);
98 
99  // Vendor ID
100  sendPulseDistanceWidthData(KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_BIT_MARK, KASEIKYO_ZERO_SPACE, aVendorCode,
101  KASEIKYO_VENDOR_ID_BITS, PROTOCOL_IS_LSB_FIRST);
102 
103  // Vendor Parity
104  uint8_t tVendorParity = aVendorCode ^ (aVendorCode >> 8);
105  tVendorParity = (tVendorParity ^ (tVendorParity >> 4)) & 0xF;
106 
107  LongUnion tSendValue;
108  tSendValue.UWord.LowWord = aAddress << KASEIKYO_VENDOR_ID_PARITY_BITS;
109  tSendValue.UByte.LowByte |= tVendorParity; // set low nibble to parity
110  tSendValue.UByte.MidHighByte = aCommand;
111  tSendValue.UByte.HighByte = aCommand ^ tSendValue.UByte.LowByte ^ tSendValue.UByte.MidLowByte; // Parity
112 
113  // Send address (device and subdevice) + command + parity + Stop bit
114  sendPulseDistanceWidthData(KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_BIT_MARK, KASEIKYO_ZERO_SPACE, tSendValue.ULong,
115  KASEIKYO_ADDRESS_BITS + KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
116 
117  tNumberOfCommands--;
118  // skip last delay!
119  if (tNumberOfCommands > 0) {
120  // send repeated command in a fixed raster
121  delay(KASEIKYO_REPEAT_SPACE / 1000);
122  }
123  }
124 }

130 /*
131  * Tested with my Panasonic DVD/TV remote
132  */
133 bool IRrecv::decodeKaseikyo() {
134 
135  decode_type_t tProtocol;
136  // Check we have enough data (100)- +4 for initial gap, start bit mark and space + stop bit mark
137  if (decodedIRData.rawDataPtr->rawlen != ((2 * KASEIKYO_BITS) + 4)) {
138  return false;
139  }
140 
141  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], KASEIKYO_HEADER_MARK)) {
142  DBG_PRINT("Kaseikyo: ");
143  DBG_PRINTLN("Header mark length is wrong");
144  return false;
145  }
146 
147  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[2], KASEIKYO_HEADER_SPACE)) {
148  DBG_PRINT("Kaseikyo: ");
149  DBG_PRINTLN("Header space length is wrong");
150  return false;
151  }
152 
153  // decode Vendor ID
154  if (!decodePulseDistanceData(KASEIKYO_VENDOR_ID_BITS, 3, KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE, KASEIKYO_ZERO_SPACE, false)) {
155  DBG_PRINT("Kaseikyo: ");
156  DBG_PRINTLN("Vendor ID decode failed");
157  return false;
158  }
159 
160  uint16_t tVendorId = decodedIRData.decodedRawData;
161  if (tVendorId == PANASONIC_VENDOR_ID_CODE) {
162  tProtocol = PANASONIC;
163  } else if (tVendorId == SHARP_VENDOR_ID_CODE) {
164  tProtocol = KASEIKYO_SHARP;
165  } else if (tVendorId == DENON_VENDOR_ID_CODE) {
166  tProtocol = KASEIKYO_DENON;
167  } else if (tVendorId == JVC_VENDOR_ID_CODE) {
168  tProtocol = KASEIKYO_JVC;
169  } else if (tVendorId == MITSUBISHI_VENDOR_ID_CODE) {
170  tProtocol = KASEIKYO_MITSUBISHI;
171  } else {
172  tProtocol = KASEIKYO;
173  }
174 
175  // Vendor Parity
176  uint8_t tVendorParity = tVendorId ^ (tVendorId >> 8);
177  tVendorParity = (tVendorParity ^ (tVendorParity >> 4)) & 0xF;
178 
179  // decode address (device and subdevice) + command + parity
180  if (!decodePulseDistanceData(
181  KASEIKYO_VENDOR_ID_PARITY_BITS + KASEIKYO_ADDRESS_BITS + KASEIKYO_COMMAND_BITS + KASEIKYO_PARITY_BITS,
182  3 + (2 * KASEIKYO_VENDOR_ID_BITS), KASEIKYO_BIT_MARK, KASEIKYO_ONE_SPACE,
183  KASEIKYO_ZERO_SPACE, false)) {
184  DBG_PRINT("Kaseikyo: ");
185  DBG_PRINTLN("Address, command + parity decode failed");
186  return false;
187  }
188 
189  // Success
190 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
191  LongUnion tValue;
192  tValue.ULong = decodedIRData.decodedRawData;
193  decodedIRData.address = (tValue.UWord.LowWord >> KASEIKYO_VENDOR_ID_PARITY_BITS); // remove vendor parity
194  decodedIRData.command = tValue.UByte.MidHighByte;
195  uint8_t tParity = tValue.UByte.LowByte ^ tValue.UByte.MidLowByte ^ tValue.UByte.MidHighByte;
196 
197  if (tVendorParity != (tValue.UByte.LowByte & 0xF)) {
198  DBG_PRINT("Kaseikyo: ");
199  DBG_PRINT("4 bit VendorID Parity is not correct. expected=0x");
200  DBG_PRINT(tVendorParity, HEX);
201  DBG_PRINT(" received=0x");
202  DBG_PRINT(decodedIRData.decodedRawData, HEX);
203  DBG_PRINT(" VendorID=0x");
204  DBG_PRINTLN(tVendorId, HEX);
205  decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
206  }
207 
208  if (tProtocol == KASEIKYO) {
209  decodedIRData.flags |= IRDATA_FLAGS_EXTRA_INFO;
210  // Include vendor ID in address
211  decodedIRData.extra |= tVendorId;
212  }
213 
214  if (tValue.UByte.HighByte != tParity) {
215  DBG_PRINT("Kaseikyo: ");
216  DBG_PRINT("8 bit Parity is not correct. expected=0x");
217  DBG_PRINT(tParity, HEX);
218  DBG_PRINT(" received=0x");
219  DBG_PRINT(decodedIRData.decodedRawData >> KASEIKYO_COMMAND_BITS, HEX);
220  DBG_PRINT(" address=0x");
221  DBG_PRINT(decodedIRData.address, HEX);
222  DBG_PRINT(" command=0x");
223  DBG_PRINTLN(decodedIRData.command, HEX);
224  decodedIRData.flags |= IRDATA_FLAGS_PARITY_FAILED;
225  }
226 
227  // check for repeat
228  if (decodedIRData.rawDataPtr->rawbuf[0] < (KASEIKYO_REPEAT_PERIOD / MICROS_PER_TICK)) {
229  decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
230  }
231 
232  decodedIRData.protocol = tProtocol;
233 
234  decodedIRData.numberOfBits = KASEIKYO_BITS;
235 
236  return true;
237 }
