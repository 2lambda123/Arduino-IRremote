41 // L GGGG
42 // L G
43 // L G GG
44 // L G G
45 // LLLLL GGG
46 //==============================================================================
47 // LG originally added by Darryl Smith (based on the JVC protocol)
48 // see: https://github.com/Arduino-IRremote/Arduino-IRremote/tree/master/examples/LGAirConditionerSendDemo
49 // see: https://www.mikrocontroller.net/articles/IRMP_-_english#LGAIR
50 // MSB first, timing and repeat is like NEC but 28 data bits
51 // MSB! first, 1 start bit + 8 bit address + 16 bit command + 4 bit parity + 1 stop bit.
52 // In https://github.com/Arduino-IRremote/Arduino-IRremote/discussions/755 we saw no key repetition
53 // and a intended parity error, or something I do not understand.
54 #define LG_ADDRESS_BITS 8
55 #define LG_COMMAND_BITS 16
56 #define LG_CHECKSUM_BITS 4
57 #define LG_BITS (LG_ADDRESS_BITS + LG_COMMAND_BITS + LG_CHECKSUM_BITS) // 28
58 
59 #define LG_UNIT 560 // like NEC
60 
61 #define LG_HEADER_MARK (16 * LG_UNIT) // 9000
62 #define LG_HEADER_SPACE (8 * LG_UNIT) // 4500
63 
64 #define LG_BIT_MARK LG_UNIT
65 #define LG_ONE_SPACE (3 * LG_UNIT) // 1690
66 #define LG_ZERO_SPACE LG_UNIT
67 
68 #define LG_REPEAT_HEADER_SPACE (4 * LG_UNIT) // 2250
69 #define LG_AVERAGE_DURATION 58000 // LG_HEADER_MARK + LG_HEADER_SPACE + 32 * 2,5 * LG_UNIT) + LG_UNIT // 2.5 because we assume more zeros than ones
70 #define LG_REPEAT_DURATION (LG_HEADER_MARK + LG_REPEAT_HEADER_SPACE + LG_BIT_MARK)
71 #define LG_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
72 #define LG_REPEAT_SPACE (LG_REPEAT_PERIOD - LG_AVERAGE_DURATION) // 52 ms

86 /*
87  * Repeat commands should be sent in a 110 ms raster.
88  * There is NO delay after the last sent repeat!
89  */
90 void IRsend::sendLG(uint8_t aAddress, uint16_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
91  uint32_t tRawData = ((uint32_t) aAddress << (LG_COMMAND_BITS + LG_CHECKSUM_BITS)) | ((uint32_t) aCommand << LG_CHECKSUM_BITS);
92  /*
93  * My guess of the checksum
94  */
95  uint8_t tChecksum = 0;
96  uint16_t tTempForChecksum = aCommand;
97  for (int i = 0; i < 4; ++i) {
98  tChecksum += tTempForChecksum & 0xF; // add low nibble
99  tTempForChecksum >>= 4; // shift by a nibble
100  }
101  tRawData |= (tChecksum & 0xF);
102  sendLGRaw(tRawData, aNumberOfRepeats, aIsRepeat);
103 }

105 /*
106  * Here you can put your raw data, even one with "wrong" parity
107  */
108 void IRsend::sendLGRaw(uint32_t aRawData, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
109  if (aIsRepeat) {
110  sendLGRepeat();
111  return;
112  }
113  // Set IR carrier frequency
114  enableIROut(38);
115 
116  // Header
117  mark(LG_HEADER_MARK);
118  space(LG_HEADER_SPACE);
119 
120  // MSB first
121  sendPulseDistanceWidthData(LG_BIT_MARK, LG_ONE_SPACE, LG_BIT_MARK, LG_ZERO_SPACE, aRawData, LG_BITS, PROTOCOL_IS_MSB_FIRST,
122  SEND_STOP_BIT);
123 
124  for (uint_fast8_t i = 0; i < aNumberOfRepeats; ++i) {
125  // send repeat in a 110 ms raster
126  if (i == 0) {
127  delay(LG_REPEAT_SPACE / 1000);
128  } else {
129  delay((LG_REPEAT_PERIOD - LG_REPEAT_DURATION) / 1000);
130  }
131  // send repeat
132  sendLGRepeat();
133  }
134 }

139 /*
140  * First check for right data length
141  * Next check start bit
142  * Next try the decode
143  * Last check stop bit
144  */
145 bool IRrecv::decodeLG() {
146 
147  // Check we have the right amount of data (60). The +4 is for initial gap, start bit mark and space + stop bit mark.
148  if (decodedIRData.rawDataPtr->rawlen != ((2 * LG_BITS) + 4) && (decodedIRData.rawDataPtr->rawlen != 4)) {
149  DBG_PRINT(F("LG: "));
150  DBG_PRINT("Data length=");
151  DBG_PRINT(decodedIRData.rawDataPtr->rawlen);
152  DBG_PRINTLN(" is not 60 or 4");
153  return false;
154  }
155 
156  // Check header "mark" this must be done for repeat and data
157  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], LG_HEADER_MARK)) {
158  return false;
159  }
160 
161  // Check for repeat - here we have another header space length
162  if (decodedIRData.rawDataPtr->rawlen == 4) {
163  if (matchSpace(decodedIRData.rawDataPtr->rawbuf[2], LG_REPEAT_HEADER_SPACE)
164  && matchMark(decodedIRData.rawDataPtr->rawbuf[3], LG_BIT_MARK)) {
165  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_MSB_FIRST;
166  decodedIRData.address = lastDecodedAddress;
167  decodedIRData.command = lastDecodedCommand;
168 // decodedIRData.protocol = LG; do not set it, because it can also be an NEC repeat
169  return true;
170  }
171  return false;
172  }
173 
174  // Check command header space
175  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], LG_HEADER_SPACE)) {
176  DBG_PRINT(F("LG: "));
177  DBG_PRINTLN(F("Header space length is wrong"));
178  return false;
179  }
180 
181  if (!decodePulseDistanceData(LG_BITS, 3, LG_BIT_MARK, LG_ONE_SPACE, LG_ZERO_SPACE, true)) {
182  DBG_PRINT(F("LG: "));
183  DBG_PRINTLN(F("Decode failed"));
184  return false;
185  }
186 
187  // Stop bit
188  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * LG_BITS)], LG_BIT_MARK)) {
189  DBG_PRINT(F("LG: "));
190  DBG_PRINTLN(F("Stop bit mark length is wrong"));
191  return false;
192  }
193 
194  // Success
195  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
196  decodedIRData.command = (decodedIRData.decodedRawData >> LG_CHECKSUM_BITS) & 0xFFFF;
197  decodedIRData.address = decodedIRData.decodedRawData >> (LG_COMMAND_BITS + LG_CHECKSUM_BITS); // first 8 bit
198 
199  /*
200  * My guess of the checksum
201  */
202  uint8_t tChecksum = 0;
203  uint16_t tTempForChecksum = decodedIRData.command;
204  for (int i = 0; i < 4; ++i) {
205  tChecksum += tTempForChecksum & 0xF; // add low nibble
206  tTempForChecksum >>= 4; // shift by a nibble
207  }
208  // Parity check
209  if ((tChecksum & 0xF) != (decodedIRData.decodedRawData & 0xF)) {
210  DBG_PRINT(F("LG: "));
211  DBG_PRINT("4 bit checksum is not correct. expected=0x");
212  DBG_PRINT(tChecksum, HEX);
213  DBG_PRINT(" received=0x");
214  DBG_PRINT((decodedIRData.decodedRawData & 0xF), HEX);
215  DBG_PRINT(" data=0x");
216  DBG_PRINTLN(decodedIRData.command, HEX);
217  decodedIRData.flags |= IRDATA_FLAGS_PARITY_FAILED;
218  }
219 
220  decodedIRData.protocol = LG;
221  decodedIRData.numberOfBits = LG_BITS;
222 
223  return true;
224 }

229
