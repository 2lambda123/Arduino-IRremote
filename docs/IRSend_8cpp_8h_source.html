271 void IRsend::sendPulseDistanceWidthData(unsigned int aOneMarkMicros, unsigned int aOneSpaceMicros, unsigned int aZeroMarkMicros,
272  unsigned int aZeroSpaceMicros, uint32_t aData, uint8_t aNumberOfBits, bool aMSBfirst, bool aSendStopBit) {
273 
274  if (aMSBfirst) { // Send the MSB first.
275  // send data from MSB to LSB until mask bit is shifted out
276  for (uint32_t tMask = 1UL << (aNumberOfBits - 1); tMask; tMask >>= 1) {
277  if (aData & tMask) {
278  TRACE_PRINT('1');
279  mark(aOneMarkMicros);
280  space(aOneSpaceMicros);
281  } else {
282  TRACE_PRINT('0');
283  mark(aZeroMarkMicros);
284  space(aZeroSpaceMicros);
285  }
286  }
287  } else { // Send the Least Significant Bit (LSB) first / MSB last.
288  for (uint_fast8_t bit = 0; bit < aNumberOfBits; bit++, aData >>= 1)
289  if (aData & 1) { // Send a 1
290  TRACE_PRINT('1');
291  mark(aOneMarkMicros);
292  space(aOneSpaceMicros);
293  } else { // Send a 0
294  TRACE_PRINT('0');
295  mark(aZeroMarkMicros);
296  space(aZeroSpaceMicros);
297  }
298  }
299  if (aSendStopBit) {
300  TRACE_PRINT('S');
301  mark(aZeroMarkMicros); // seems like this is used for stop bits
302  ledOff(); // Always end with the LED off
303  }
304  TRACE_PRINTLN("");
305 }

307 /*
308  * Sends Biphase data MSB first
309  * Always send start bit, do not send the trailing space of the start bit
310  * 0 -> mark+space
311  * 1 -> space+mark
312  * The output always ends with a space
313  */
314 void IRsend::sendBiphaseData(unsigned int aBiphaseTimeUnit, uint32_t aData, uint_fast8_t aNumberOfBits) {
315 
316 // do not send the trailing space of the start bit
317  mark(aBiphaseTimeUnit);
318 
319  TRACE_PRINT('S');
320  uint8_t tLastBitValue = 1; // Start bit is a 1
321 
322 // Data - Biphase code MSB first
323  for (uint32_t tMask = 1UL << (aNumberOfBits - 1); tMask; tMask >>= 1) {
324  if (aData & tMask) {
325  TRACE_PRINT('1');
326  space(aBiphaseTimeUnit);
327  mark(aBiphaseTimeUnit);
328  tLastBitValue = 1;
329 
330  } else {
331  TRACE_PRINT('0');
332 #ifdef USE_SOFT_SEND_PWM
333  (void)tLastBitValue; // to avoid compiler warnings
334  mark(aBiphaseTimeUnit); // can not eventually delay here, we must call mark to generate the signal
335 #else
336  if (tLastBitValue) {
337  // Extend the current mark in order to generate a continuous signal without short breaks
338  delayMicroseconds(aBiphaseTimeUnit);
339  } else {
340  mark(aBiphaseTimeUnit);
341  }
342 #endif
343  space(aBiphaseTimeUnit);
344  tLastBitValue = 0;
345  }
346  }
347 // ledOff(); // Always end with the LED off
348  TRACE_PRINTLN("");
349 }

357 void IRsend::mark(unsigned int aMarkMicros) {
358  setFeedbackLED(true);
359 
360 #if defined(USE_SOFT_SEND_PWM) && !defined(ESP32) // for esp32 we use PWM generation by hw_timer_t for each pin
361  unsigned long start = micros();
362  unsigned long nextPeriodEnding = start;
363  while (micros() - start < aMarkMicros) {
364  // Output the PWM pulse
365  noInterrupts(); // do not let interrupts extend the short on period
366 # ifdef SENDPIN_ON
367  SENDPIN_ON(sendPin);
368 # else
369  digitalWrite(sendPin, HIGH);
370 # endif
371  delayMicroseconds(periodOnTimeMicros); // this is normally implemented by a blocking wait
372 
373  // Output the PWM pause
374 # ifdef SENDPIN_OFF
375  SENDPIN_OFF(sendPin);
376 # else
377  digitalWrite(sendPin, LOW);
378 # endif
379  interrupts(); // Enable interrupts -to keep micros correct- for the longer off period 3.4 us until receive ISR is active (for 7 us + pop's)
380  nextPeriodEnding += periodTimeMicros;
381  while (micros() < nextPeriodEnding){
382  ;
383  }
384  }
385 
386 #else
387 # if defined(USE_NO_SEND_PWM)
388 # ifdef SENDPIN_OFF
389  SENDPIN_OFF(sendPin);
390 # else
391  digitalWrite(sendPin, LOW); // Set output to active low.
392 # endif
393 
394 # else
395  TIMER_ENABLE_SEND_PWM; // Enable pin 3 PWM output
396 # endif // USE_SOFT_SEND_PWM
397 
398  customDelayMicroseconds(aMarkMicros);
399  ledOff();
400 #endif // USE_SOFT_SEND_PWM
401
