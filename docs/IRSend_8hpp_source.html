345 void IRsend::sendPulseDistanceWidthData(unsigned int aOneMarkMicros, unsigned int aOneSpaceMicros, unsigned int aZeroMarkMicros,
346  unsigned int aZeroSpaceMicros, uint32_t aData, uint_fast8_t aNumberOfBits, bool aMSBfirst, bool aSendStopBit) {
347 
348  if (aMSBfirst) { // Send the MSB first.
349  // send data from MSB to LSB until mask bit is shifted out
350  for (uint32_t tMask = 1UL << (aNumberOfBits - 1); tMask; tMask >>= 1) {
351  if (aData & tMask) {
352  IR_TRACE_PRINT('1');
353  mark(aOneMarkMicros);
354  space(aOneSpaceMicros);
355  } else {
356  IR_TRACE_PRINT('0');
357  mark(aZeroMarkMicros);
358  space(aZeroSpaceMicros);
359  }
360  }
361  } else { // Send the Least Significant Bit (LSB) first / MSB last.
362  for (uint_fast8_t bit = 0; bit < aNumberOfBits; bit++, aData >>= 1)
363  if (aData & 1) { // Send a 1
364  IR_TRACE_PRINT('1');
365  mark(aOneMarkMicros);
366  space(aOneSpaceMicros);
367  } else { // Send a 0
368  IR_TRACE_PRINT('0');
369  mark(aZeroMarkMicros);
370  space(aZeroSpaceMicros);
371  }
372  }
373  if (aSendStopBit) {
374  IR_TRACE_PRINT('S');
375  mark(aZeroMarkMicros); // seems like this is used for stop bits
376  }
377  IR_TRACE_PRINTLN(F(""));
378 }

388 void IRsend::sendBiphaseData(unsigned int aBiphaseTimeUnit, uint32_t aData, uint_fast8_t aNumberOfBits) {
389 
390  IR_TRACE_PRINT(F("0x"));
391  IR_TRACE_PRINT(aData, HEX);
392 
393  IR_TRACE_PRINT(F(" S"));
394 
395  // Data - Biphase code MSB first
396  // prepare for start with sending the start bit, which is 1
397  uint32_t tMask = 1UL << aNumberOfBits; // mask is now set for the virtual start bit
398  uint_fast8_t tLastBitValue = 1; // Start bit is a 1
399  bool tNextBitIsOne = 1; // Start bit is a 1
400  for (uint_fast8_t i = aNumberOfBits + 1; i > 0; i--) {
401  bool tCurrentBitIsOne = tNextBitIsOne;
402  tMask >>= 1;
403  tNextBitIsOne = ((aData & tMask) != 0) || (i == 1); // true for last bit to avoid extension of mark
404  if (tCurrentBitIsOne) {
405  IR_TRACE_PRINT('1');
406  space(aBiphaseTimeUnit);
407  if (tNextBitIsOne) {
408  mark(aBiphaseTimeUnit);
409  } else {
410  // if next bit is 0, extend the current mark in order to generate a continuous signal without short breaks
411  mark(2 * aBiphaseTimeUnit);
412  }
413  tLastBitValue = 1;
414 
415  } else {
416  IR_TRACE_PRINT('0');
417  if (!tLastBitValue) {
418  mark(aBiphaseTimeUnit);
419  }
420  space(aBiphaseTimeUnit);
421  tLastBitValue = 0;
422  }
423  }
424  IR_TRACE_PRINTLN(F(""));
425 }

433 void IRsend::mark(unsigned int aMarkMicros) {
434 
435 #if defined(SEND_PWM_BY_TIMER)
436 # if !defined(NO_LED_FEEDBACK_CODE)
437  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
438  setFeedbackLED(true);
439  }
440 # endif
441  ENABLE_SEND_PWM_BY_TIMER; // Enable timer or ledcWrite() generated PWM output
442  customDelayMicroseconds(aMarkMicros);
443  IRLedOff(); // manages also feedback LED
444 
445 #elif defined(USE_NO_SEND_PWM)
446 # if !defined(NO_LED_FEEDBACK_CODE)
447  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
448  setFeedbackLED(true);
449  }
450 # endif
451 # if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && !defined(OUTPUT_OPEN_DRAIN)
452  pinModeFast(sendPin, OUTPUT); // active state for mimicking open drain
453 # else
454  digitalWriteFast(sendPin, LOW); // Set output to active low.
455 # endif
456 
457  customDelayMicroseconds(aMarkMicros);
458  IRLedOff();
459 # if !defined(NO_LED_FEEDBACK_CODE)
460  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
461  setFeedbackLED(false);
462  }
463 # endif
464 
465 #else
466  unsigned long tStartMicros = micros();
467  unsigned long tNextPeriodEnding = tStartMicros;
468  unsigned long tMicros;
469 # if !defined(NO_LED_FEEDBACK_CODE)
470  bool FeedbackLedIsActive = false;
471 # endif
472 
473  do {
474 // digitalToggleFast(_IR_TIMING_TEST_PIN);
475  /*
476  * Output the PWM pulse
477  */
478  noInterrupts(); // do not let interrupts extend the short on period
479 # if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN)
480 # if defined(OUTPUT_OPEN_DRAIN)
481  digitalWriteFast(sendPin, LOW); // active state for open drain
482 # else
483  pinModeFast(sendPin, OUTPUT); // active state for mimicking open drain
484 # endif
485 # else
486  // 3.5 us from FeedbackLed on to pin setting. 5.7 us from call of mark() to pin setting incl. setting of feedback pin.
487  // 4.3 us from do{ to pin setting if sendPin is no constant
488  digitalWriteFast(sendPin, HIGH);
489 # endif
490  delayMicroseconds(periodOnTimeMicros); // this is normally implemented by a blocking wait
491 
492  /*
493  * Output the PWM pause
494  */
495 # if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && !defined(OUTPUT_OPEN_DRAIN)
496 # if defined(OUTPUT_OPEN_DRAIN)
497  digitalWriteFast(sendPin, HIGH); // Set output with pin mode OUTPUT_OPEN_DRAIN to inactive high.
498 # else
499  pinModeFast(sendPin, INPUT); // to mimic the open drain inactive state
500 # endif
501 
502 # else
503  digitalWriteFast(sendPin, LOW);
504 # endif
505  interrupts(); // Enable interrupts - to keep micros correct- for the longer off period 3.4 us until receive ISR is active (for 7 us + pop's)
506 
507 # if !defined(NO_LED_FEEDBACK_CODE)
508  /*
509  * Delayed call of setFeedbackLED() to get better timing
510  */
511  if (!FeedbackLedIsActive) {
512  FeedbackLedIsActive = true;
513  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
514  setFeedbackLED(true);
515  }
516  }
517 # endif
518  /*
519  * PWM pause timing
520  */
521  tNextPeriodEnding += periodTimeMicros;
522  do {
523  tMicros = micros(); // we have only 4 us resolution for AVR @16MHz
524  /*
525  * Exit the forever loop if aMarkMicros has reached
526  */
527  unsigned int tDeltaMicros = tMicros - tStartMicros;
528 #if defined(__AVR__)
529 // tDeltaMicros += (160 / CLOCKS_PER_MICRO); // adding this once increases program size !
530 # if !defined(NO_LED_FEEDBACK_CODE)
531  if (tDeltaMicros >= aMarkMicros - (30 + (112 / CLOCKS_PER_MICRO))) { // 30 to be constant. Using periodTimeMicros increases program size too much.
532  // reset feedback led in the last pause before end
533  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
534  setFeedbackLED(false);
535  }
536  }
537 # endif
538  if (tDeltaMicros >= aMarkMicros - (112 / CLOCKS_PER_MICRO)) { // To compensate for call duration - 112 is an empirical value
539 #else
540  if (tDeltaMicros >= aMarkMicros) {
541 # if !defined(NO_LED_FEEDBACK_CODE)
542  if (FeedbackLEDControl.LedFeedbackEnabled == LED_FEEDBACK_ENABLED_FOR_SEND) {
543  setFeedbackLED(false);
544  }
545 # endif
546 #endif
547  return;
548  }
549 // digitalToggleFast(_IR_TIMING_TEST_PIN); // 3.0 us per call @16MHz
550  }while (tMicros < tNextPeriodEnding); // 3.4 us @16MHz
551  }while (true);
552 # endif
553 }

617 void IRsend::enableIROut(uint_fast8_t aFrequencyKHz) {
618 #if defined(SEND_PWM_BY_TIMER)
619  timerConfigForSend(aFrequencyKHz); // must set output pin mode and disable receive interrupt if required, e.g. uses the same resource
620 
621 #elif defined(USE_NO_SEND_PWM)
622  (void) aFrequencyKHz;
623 
624 #else
625  periodTimeMicros = (1000U + (aFrequencyKHz / 2)) / aFrequencyKHz; // rounded value -> 26 for 38.46 kHz, 27 for 37.04 kHz, 25 for 40 kHz.
626 # if defined(IR_SEND_PIN)
627  periodOnTimeMicros = (((periodTimeMicros * IR_SEND_DUTY_CYCLE_PERCENT) + 50) / 100U); // +50 for rounding -> 830/100 for 30% and 16 MHz
628 # else
629  // Heuristics! We require a nanosecond correction for "slow" digitalWrite() functions
630  periodOnTimeMicros = (((periodTimeMicros * IR_SEND_DUTY_CYCLE_PERCENT) + 50 - (PULSE_CORRECTION_NANOS / 10)) / 100U); // +50 for rounding -> 530/100 for 30% and 16 MHz
631 # endif
632 #endif // defined(SEND_PWM_BY_TIMER)
633 
634 #if defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN) && defined(OUTPUT_OPEN_DRAIN) // the mode INPUT for mimicking open drain is set at IRLedOff()
635 # if defined(IR_SEND_PIN)
636  pinModeFast(IR_SEND_PIN, OUTPUT_OPEN_DRAIN);
637 # else
638  pinModeFast(sendPin, OUTPUT_OPEN_DRAIN);
639 # endif
640 #else
641 
642  // For Non AVR platforms pin mode for SEND_PWM_BY_TIMER must be handled by the timerConfigForSend() function
643  // because ESP 2.0.2 ledcWrite does not work if pin mode is set, and RP2040 requires gpio_set_function(IR_SEND_PIN, GPIO_FUNC_PWM);
644 # if defined(__AVR__) || !defined(SEND_PWM_BY_TIMER)
645 # if defined(IR_SEND_PIN)
646  pinModeFast(IR_SEND_PIN, OUTPUT);
647 # else
648  pinModeFast(sendPin, OUTPUT);
649 # endif
650 # endif
651 #endif // defined(USE_OPEN_DRAIN_OUTPUT_FOR_SEND_PIN)
652 }
