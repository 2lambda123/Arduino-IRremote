3  *
4  * Receives IR protocol data of NEC protocol using pin change interrupts.
5  * NEC is the protocol of most cheap remote controls for Arduino.
6  *
7  * No parity check is done!
8  * On a completely received IR command, the user function handleReceivedIRData(uint16_t aAddress, uint8_t aCommand, bool isRepetition)
9  * is called in Interrupt context but with interrupts being enabled to enable use of delay() etc.

76 void IRPinChangeInterruptHandler(void) {
77 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
78  digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
79 #endif
80  /*
81  * Save IR input level
82  * Negative logic, true / HIGH means inactive / IR space, LOW / false means IR mark.
83  */
84  uint_fast8_t tIRLevel = digitalReadFast(IR_INPUT_PIN);
85 
86 #if !defined(DO_NOT_USE_FEEDBACK_LED) && defined(IR_FEEDBACK_LED_PIN)
87  digitalWriteFast(IR_FEEDBACK_LED_PIN, !tIRLevel);
88 #endif
89 
90  /*
91  * 1. compute microseconds after last change
92  */
93  uint32_t tCurrentMicros = micros();
94  uint16_t tMicrosOfMarkOrSpace = tCurrentMicros - TinyIRReceiverControl.LastChangeMicros;
95  TinyIRReceiverControl.LastChangeMicros = tCurrentMicros;
96 
97  uint8_t tState = TinyIRReceiverControl.IRReceiverState;
98 
99 #ifdef TRACE
100  Serial.print(tState);
101  Serial.print(' ');
102 // Serial.print(F(" I="));
103 // Serial.print(tIRLevel);
104 // Serial.print(F(" D="));
105 // Serial.print(tDeltaMicros);
106 // Serial.println();
107 #endif
108 
109  if (tIRLevel == LOW) {
110  /*
111  * We have a mark here
112  */
113  if (tMicrosOfMarkOrSpace > 2 * NEC_HEADER_MARK) {
114  // timeout -> must reset state machine
115  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
116  }
117  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_MARK) {
118  // We are at the beginning of the header mark, check timing at the next transition
119  tState = IR_RECEIVER_STATE_WAITING_FOR_START_SPACE;
120  }
121 
122  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK) {
123  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_HEADER_SPACE)
124  && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_HEADER_SPACE)) {
125  /*
126  * We have a valid data header space here -> initialize data
127  */
128  TinyIRReceiverControl.IRRawDataBitCounter = 0;
129  TinyIRReceiverControl.IRRawData.ULong = 0;
130  TinyIRReceiverControl.IRRawDataMask = 1;
131  TinyIRReceiverControl.IRRepeatDetected = false;
132  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
133  } else if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_REPEAT_HEADER_SPACE)
134  && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_REPEAT_HEADER_SPACE)
135  && TinyIRReceiverControl.IRRawDataBitCounter >= NEC_BITS) {
136  /*
137  * We have a repeat header here and no broken receive before -> set repeat flag
138  */
139  TinyIRReceiverControl.IRRepeatDetected = true;
140  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
141  } else {
142  // This parts are optimized by the compiler into jumps to one code :-)
143  // Wrong length -> reset state
144  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
145  }
146  }
147 
148  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK) {
149  // Check data space length
150  if (tMicrosOfMarkOrSpace >= lowerValue(NEC_ZERO_SPACE) && tMicrosOfMarkOrSpace <= upperValue(NEC_ONE_SPACE)) {
151  // We have a valid bit here
152  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE;
153  if (tMicrosOfMarkOrSpace >= 2 * NEC_UNIT) {
154  // we received a 1
155  TinyIRReceiverControl.IRRawData.ULong |= TinyIRReceiverControl.IRRawDataMask;
156  } else {
157  // we received a 0 - empty code for documentation
158  }
159  // prepare for next bit
160  TinyIRReceiverControl.IRRawDataMask = TinyIRReceiverControl.IRRawDataMask << 1;
161  TinyIRReceiverControl.IRRawDataBitCounter++;
162  } else {
163  // Wrong length -> reset state
164  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
165  }
166  } else {
167  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
168  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
169  }
170  }
171 
172  else {
173  /*
174  * We have a space here
175  */
176  if (tState == IR_RECEIVER_STATE_WAITING_FOR_START_SPACE) {
177  /*
178  * Check length of header mark here
179  */
180  if (tMicrosOfMarkOrSpace >= lowerValue25Percent(NEC_HEADER_MARK)
181  && tMicrosOfMarkOrSpace <= upperValue25Percent(NEC_HEADER_MARK)) {
182  tState = IR_RECEIVER_STATE_WAITING_FOR_FIRST_DATA_MARK;
183  } else {
184  // Wrong length of header mark -> reset state
185  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
186  }
187  }
188 
189  else if (tState == IR_RECEIVER_STATE_WAITING_FOR_DATA_SPACE) {
190  // Check data mark length
191  if (tMicrosOfMarkOrSpace >= lowerValue(NEC_BIT_MARK) && tMicrosOfMarkOrSpace <= upperValue(NEC_BIT_MARK)) {
192  /*
193  * We have a valid mark here, check for transmission complete
194  */
195  if (TinyIRReceiverControl.IRRawDataBitCounter >= NEC_BITS || TinyIRReceiverControl.IRRepeatDetected) {
196  /*
197  * Code complete -> call callback, no parity check!
198  */
199  // Reset state for new start
200  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
201 #if !defined(ARDUINO_ARCH_MBED)
202  interrupts();
203 #endif
204  /*
205  * Address reduction to 8 bit
206  */
207  if (TinyIRReceiverControl.IRRawData.UByte.LowByte
208  == (uint8_t) (~TinyIRReceiverControl.IRRawData.UByte.MidLowByte)) {
209  // standard 8 bit address NEC protocol
210  TinyIRReceiverControl.IRRawData.UByte.MidLowByte = 0; // Address is the first 8 bit
211  }
212 
213  /*
214  * Call user provided callback here
215  */
216  handleReceivedTinyIRData(TinyIRReceiverControl.IRRawData.UWord.LowWord,
217  TinyIRReceiverControl.IRRawData.UByte.MidHighByte, TinyIRReceiverControl.IRRepeatDetected);
218 
219  } else {
220  // not finished yet
221  tState = IR_RECEIVER_STATE_WAITING_FOR_DATA_MARK;
222  }
223  } else {
224  // Wrong length -> reset state
225  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
226  }
227  } else {
228  // error wrong state for the received level, e.g. if we missed one change interrupt -> reset state
229  tState = IR_RECEIVER_STATE_WAITING_FOR_START_MARK;
230  }
231  }
232 
233  TinyIRReceiverControl.IRReceiverState = tState;
234 #ifdef IR_MEASURE_TIMING
235  digitalWriteFast(IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
236 #endif
237 }
