65 bool aggregateArrayCounts(uint8_t aArray[], uint8_t aMaxIndex, uint8_t *aShortIndex, uint8_t *aLongIndex) {
66  uint8_t tSum = 0;
67  uint16_t tWeightedSum = 0;
68  for (uint_fast8_t i = 0; i <= aMaxIndex; i++) {
69  uint8_t tCurrentDurations = aArray[i];
70  if (tCurrentDurations != 0) {
71  // Add it to sum and remove array content
72  tSum += tCurrentDurations;
73  tWeightedSum += (tCurrentDurations * i);
74  aArray[i] = 0;
75  }
76  if ((tCurrentDurations == 0 || i == aMaxIndex) && tSum != 0) {
77  // here we have a sum and a gap after the values
78  uint8_t tAggregateIndex = (tWeightedSum + (tSum / 2)) / tSum; // with rounding
79  aArray[tAggregateIndex] = tSum; // disabling this line increases code size by 2 - unbelievable!
80  // store aggregate for later decoding
81  if (*aShortIndex == 0) {
82  *aShortIndex = tAggregateIndex;
83  } else if (*aLongIndex == 0) {
84  *aLongIndex = tAggregateIndex;
85  } else {
86  // we have 3 bins => this is likely no pulse width or distance protocol. e.g. it can be RC5.
87  return false;
88  }
89  // initialize for next aggregation
90  tSum = 0;
91  tWeightedSum = 0;
92  }
93  }
94  return true;
95 }

97 /*
98  * Try to decode a pulse width or pulse distance protocol.
99  * 1. analyze all space and mark length
100  * 2. decide if we have an pulse width or distance protocol
101  * 3. try to decode with the mark and space data found in step 1.
102  * No data and address, only raw data as result.
103  */
104 bool IRrecv::decodeDistance() {
105  uint8_t tDurationArray[DURATION_ARRAY_SIZE];
106 
107  /*
108  * Accept only protocols with at least 8 bits
109  */
110  if (decodedIRData.rawDataPtr->rawlen < (2 * 8) + 4) {
111  DEBUG_PRINT("PULSE_DISTANCE: ");
112  DEBUG_PRINT("Data length=");
113  DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
114  DEBUG_PRINTLN(" is less than 20");
115  return false;
116  }
117 
118  uint_fast8_t i;
119 // Reset array
120  memset(tDurationArray, 0, sizeof(tDurationArray));
121 
122  uint8_t tMaxDurationIndex = 0;
123  // Count space durations. Skip leading start and trailing stop bit.
124  for (i = 4; i < decodedIRData.rawDataPtr->rawlen - 2; i += 2) {
125  uint8_t tDurationTicks = decodedIRData.rawDataPtr->rawbuf[i];
126  if (tDurationTicks < sizeof(tDurationArray)) {
127  tDurationArray[tDurationTicks]++;
128  if (tMaxDurationIndex < tDurationTicks) {
129  tMaxDurationIndex = tDurationTicks;
130  }
131  }
132  }
133 
134 // aggregate counts to one duration bin
135  uint8_t tSpaceTicksShort = 0;
136  uint8_t tSpaceTicksLong = 0;
137  if (!aggregateArrayCounts(tDurationArray, tMaxDurationIndex, &tSpaceTicksShort, &tSpaceTicksLong)) {
138  DEBUG_PRINT(F("PULSE_DISTANCE: "));
139  DEBUG_PRINTLN(F("Space aggregation failed, more than 2 distinct duration values found"));
140  return false;
141  }
142 
143 #if defined(DEBUG)
144  Serial.println(F("Space:"));
145  printDurations(tDurationArray, tMaxDurationIndex);
146 #endif
147 
148  // Reset array
149  memset(tDurationArray, 0, sizeof(tDurationArray));
150 
151  tMaxDurationIndex = 0;
152  // Count mark durations. Skip leading start and trailing stop bit.
153  for (i = 3; i < decodedIRData.rawDataPtr->rawlen - 2; i += 2) {
154  uint8_t tDurationTicks = decodedIRData.rawDataPtr->rawbuf[i];
155  if (tDurationTicks < sizeof(tDurationArray)) {
156  tDurationArray[tDurationTicks]++;
157  if (tMaxDurationIndex < tDurationTicks) {
158  tMaxDurationIndex = tDurationTicks;
159  }
160  }
161  }
162 
163  uint8_t tMarkTicksShort = 0;
164  uint8_t tMarkTicksLong = 0;
165  if (!aggregateArrayCounts(tDurationArray, tMaxDurationIndex, &tMarkTicksShort, &tMarkTicksLong)) {
166  DEBUG_PRINT(F("PULSE_DISTANCE: "));
167  DEBUG_PRINTLN(F("Mark aggregation failed, more than 2 distinct duration values found"));
168  }
169 
170 #if defined(DEBUG)
171  Serial.println(F("Mark:"));
172  printDurations(tDurationArray, tMaxDurationIndex);
173 #endif
174  // skip leading start and trailing stop bit.
175  uint8_t tNumberOfBits = (decodedIRData.rawDataPtr->rawlen / 2) - 2;
176  uint8_t tStartIndex = 3;
177  decodedIRData.numberOfBits = tNumberOfBits;
178 
179  // adjust for longer data like Kaseikyo
180  if (tNumberOfBits > 32) {
181  tNumberOfBits = 32;
182  tStartIndex = decodedIRData.rawDataPtr->rawlen - 65;
183  }
184 
185  /*
186  * decide, if we have an pulse width or distance protocol
187  */
188  if (tSpaceTicksLong == 0) {
189  if (tMarkTicksLong == 0) {
190  DEBUG_PRINT(F("PULSE_DISTANCE: "));
191  DEBUG_PRINTLN(F("Only 1 distinct duration value for each space and mark found"));
192  return false;
193  }
194 // // check if last bit can be decoded as data or not, in this case take it as a stop bit
195 // if (decodePulseWidthData(1, decodedIRData.rawDataPtr->rawlen - 3, tMarkTicksLong * MICROS_PER_TICK,
196 // tMarkTicksShort * MICROS_PER_TICK, tSpaceTicksShort * MICROS_PER_TICK, DISTANCE_DO_MSB_DECODING)) {
197 // tNumberOfBits++;
198 // }
199  // decode without leading start bit. Currently only seen for sony protocol
200  if (!decodePulseWidthData(tNumberOfBits, tStartIndex, tMarkTicksLong * MICROS_PER_TICK,
201  tMarkTicksShort * MICROS_PER_TICK, tSpaceTicksShort * MICROS_PER_TICK, DISTANCE_DO_MSB_DECODING)) {
202  DEBUG_PRINT(F("PULSE_WIDTH: "));
203  DEBUG_PRINTLN(F("Decode failed"));
204  return false;
205  }
206  DEBUG_PRINT(F("PULSE_WIDTH: "));
207  DEBUG_PRINT(F(" OneMarkMicros="));
208  DEBUG_PRINT(tMarkTicksLong * MICROS_PER_TICK);
209  DEBUG_PRINT(F(" ZeroMarkMicros="));
210  DEBUG_PRINT(tMarkTicksShort* MICROS_PER_TICK);
211  DEBUG_PRINT(F(" ZeroSpaceMicros="));
212  DEBUG_PRINTLN(tSpaceTicksShort* MICROS_PER_TICK);
213  // Store ticks used for decoding in extra
214  decodedIRData.extra = (tMarkTicksShort << 8) | tMarkTicksLong;
215  decodedIRData.protocol = PULSE_WIDTH;
216  } else {
217 // // check if last bit can be decoded as data or not, in this case take it as a stop bit
218 // if (decodePulseDistanceData(1, decodedIRData.rawDataPtr->rawlen - 3, tMarkTicksShort * MICROS_PER_TICK,
219 // tSpaceTicksLong * MICROS_PER_TICK, tSpaceTicksShort * MICROS_PER_TICK, DISTANCE_DO_MSB_DECODING)) {
220 // Serial.print(F("tNumberOfBits++ "));
221 // tNumberOfBits++;
222 // }
223 
224  if (!decodePulseDistanceData(tNumberOfBits, tStartIndex, tMarkTicksShort * MICROS_PER_TICK,
225  tSpaceTicksLong * MICROS_PER_TICK, tSpaceTicksShort * MICROS_PER_TICK, DISTANCE_DO_MSB_DECODING)) {
226  DEBUG_PRINT(F("PULSE_DISTANCE: "));
227  DEBUG_PRINTLN(F("Decode failed"));
228  return false;
229  }
230  DEBUG_PRINT(F("PULSE_DISTANCE: "));
231  DEBUG_PRINT(F("BitMarkMicros="));
232  DEBUG_PRINT(tMarkTicksShort* MICROS_PER_TICK);
233  DEBUG_PRINT(F(" OneSpaceMicros="));
234  DEBUG_PRINT(tSpaceTicksLong* MICROS_PER_TICK);
235  DEBUG_PRINT(F(" ZeroSpaceMicros="));
236  DEBUG_PRINTLN(tSpaceTicksShort* MICROS_PER_TICK);
237  // Store ticks used for decoding in extra
238  decodedIRData.extra = (tSpaceTicksShort << 8) | tSpaceTicksLong;
239  decodedIRData.protocol = PULSE_DISTANCE;
240  }
241 
242  if (DISTANCE_DO_MSB_DECODING) {
243  decodedIRData.flags = IRDATA_FLAGS_IS_MSB_FIRST;
244  }
245 
246  return true;
247 }
