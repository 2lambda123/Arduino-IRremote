44 //==============================================================================
45 // N N EEEEE CCCC
46 // NN N E C
47 // N N N EEE C
48 // N NN E C
49 // N N EEEEE CCCC
50 //==============================================================================
51 // see: https://www.sbprojects.net/knowledge/ir/nec.php
52 // for Apple see https://en.wikipedia.org/wiki/Apple_Remote
53 // ONKYO like NEC but 16 independent command bits
54 // LSB first, 1 start bit + 16 bit address (or 8 bit address and 8 bit inverted address) + 8 bit command + 8 bit inverted command + 1 stop bit.
55 // Standard NEC send a special fixed repeat frame, but I have a DVD remote, which send the same full frame after the 110 ms.
56 //
57 #define NEC_ADDRESS_BITS 16 // 16 bit address or 8 bit address and 8 bit inverted address
58 #define NEC_COMMAND_BITS 16 // Command and inverted command
59 
60 #define NEC_BITS (NEC_ADDRESS_BITS + NEC_COMMAND_BITS)
61 #define NEC_UNIT 560 // 21.28 periods of 38 kHz, 11.2 ticks TICKS_LOW = 8.358 TICKS_HIGH = 15.0
62 
63 #define NEC_HEADER_MARK (16 * NEC_UNIT) // 9000 / 180
64 #define NEC_HEADER_SPACE (8 * NEC_UNIT) // 4500 / 90
65 
66 #define NEC_BIT_MARK NEC_UNIT
67 #define NEC_ONE_SPACE (3 * NEC_UNIT) // 1690 / 33.8 TICKS_LOW = 25.07 TICKS_HIGH = 45.0
68 #define NEC_ZERO_SPACE NEC_UNIT
69 
70 #define NEC_REPEAT_HEADER_SPACE (4 * NEC_UNIT) // 2250
71 
72 #define NEC_AVERAGE_DURATION 62000 // NEC_HEADER_MARK + NEC_HEADER_SPACE + 32 * 2,5 * NEC_UNIT + NEC_UNIT // 2.5 because we assume more zeros than ones
73 #define NEC_MINIMAL_DURATION 49900 // NEC_HEADER_MARK + NEC_HEADER_SPACE + 32 * 2 * NEC_UNIT + NEC_UNIT // 2.5 because we assume more zeros than ones
74 #define NEC_REPEAT_DURATION (NEC_HEADER_MARK + NEC_REPEAT_HEADER_SPACE + NEC_BIT_MARK) // 12 ms
75 #define NEC_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
76 #define NEC_REPEAT_SPACE (NEC_REPEAT_PERIOD - NEC_AVERAGE_DURATION) // 48 ms
77 #define NEC_MAXIMUM_REPEAT_SPACE (NEC_REPEAT_PERIOD - NEC_MINIMAL_DURATION + 5) // 65 ms
78 
79 #define APPLE_ADDRESS 0x87EE
80 //+=============================================================================

95  * Repeat commands should be sent in a 110 ms raster.
96  * There is NO delay after the last sent repeat!
97  * https://www.sbprojects.net/knowledge/ir/nec.php
98  * @param aIsRepeat if true, send only one repeat frame without leading and trailing space
99  */
100 void IRsend::sendNEC(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
101 
102  LongUnion tRawData;
103 
104  // Address 16 bit LSB first
105  if ((aAddress & 0xFF00) == 0) {
106  // assume 8 bit address -> send 8 address bits and then 8 inverted address bits LSB first
107  tRawData.UByte.LowByte = aAddress;
108  tRawData.UByte.MidLowByte = ~tRawData.UByte.LowByte;
109  } else {
110  tRawData.UWord.LowWord = aAddress;
111  }
112 
113  // send 8 command bits and then 8 inverted command bits LSB first
114  tRawData.UByte.MidHighByte = aCommand;
115  tRawData.UByte.HighByte = ~aCommand;
116 
117  sendNECRaw(tRawData.ULong, aNumberOfRepeats, aIsRepeat);
118 }

159 void IRsend::sendNECRaw(uint32_t aRawData, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
160  if (aIsRepeat) {
161  sendNECRepeat();
162  return;
163  }
164  // Set IR carrier frequency
165  enableIROut(NEC_KHZ);
166 
167  // Header
168  mark(NEC_HEADER_MARK);
169  space(NEC_HEADER_SPACE);
170 
171  // LSB first + stop bit
172  sendPulseDistanceWidthData(NEC_BIT_MARK, NEC_ONE_SPACE, NEC_BIT_MARK, NEC_ZERO_SPACE, aRawData, NEC_BITS, PROTOCOL_IS_LSB_FIRST,
173  SEND_STOP_BIT);
174 
175  for (uint_fast8_t i = 0; i < aNumberOfRepeats; ++i) {
176  // send repeat in a 110 ms raster
177  if (i == 0) {
178  delay(NEC_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
179  } else {
180  delay((NEC_REPEAT_PERIOD - NEC_REPEAT_DURATION) / MICROS_IN_ONE_MILLI);
181  }
182  // send repeat
183  sendNECRepeat();
184  }
185  IrReceiver.restartAfterSend();
186 }

199 bool IRrecv::decodeNEC() {
200 
201  // Check we have the right amount of data (68). The +4 is for initial gap, start bit mark and space + stop bit mark.
202  if (decodedIRData.rawDataPtr->rawlen != ((2 * NEC_BITS) + 4) && (decodedIRData.rawDataPtr->rawlen != 4)) {
203  IR_DEBUG_PRINT(F("NEC: "));
204  IR_DEBUG_PRINT(F("Data length="));
205  IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
206  IR_DEBUG_PRINTLN(F(" is not 68 or 4"));
207  return false;
208  }
209 
210  // Check header "mark" this must be done for repeat and data
211  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], NEC_HEADER_MARK)) {
212  return false;
213  }
214 
215  // Check for repeat - here we have another header space length
216  if (decodedIRData.rawDataPtr->rawlen == 4) {
217  if (matchSpace(decodedIRData.rawDataPtr->rawbuf[2], NEC_REPEAT_HEADER_SPACE)
218  && matchMark(decodedIRData.rawDataPtr->rawbuf[3], NEC_BIT_MARK)) {
219  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
220  decodedIRData.address = lastDecodedAddress;
221  decodedIRData.command = lastDecodedCommand;
222  decodedIRData.protocol = lastDecodedProtocol;
223  return true;
224  }
225  return false;
226  }
227 
228  // Check command header space
229  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], NEC_HEADER_SPACE)) {
230  IR_DEBUG_PRINT(F("NEC: "));
231  IR_DEBUG_PRINTLN(F("Header space length is wrong"));
232  return false;
233  }
234 
235  if (!decodePulseDistanceData(NEC_BITS, 3, NEC_BIT_MARK, NEC_ONE_SPACE, NEC_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
236  IR_DEBUG_PRINT(F("NEC: "));
237  IR_DEBUG_PRINTLN(F("Decode failed"));
238  return false;
239  }
240 
241  // Stop bit
242  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * NEC_BITS)], NEC_BIT_MARK)) {
243  IR_DEBUG_PRINT(F("NEC: "));
244  IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
245  return false;
246  }
247 
248  // Success
249 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
250  LongUnion tValue;
251  tValue.ULong = decodedIRData.decodedRawData;
252  decodedIRData.command = tValue.UByte.MidHighByte; // 8 bit
253  // Address
254  if (tValue.UWord.LowWord == APPLE_ADDRESS) {
255  /*
256  * Apple
257  */
258  decodedIRData.protocol = APPLE;
259  decodedIRData.address = tValue.UByte.HighByte;
260 
261  } else {
262  /*
263  * NEC LSB first, so first sent bit is also LSB of decodedIRData.decodedRawData
264  */
265  if (tValue.UByte.LowByte == (uint8_t) (~tValue.UByte.MidLowByte)) {
266  // standard 8 bit address NEC protocol
267  decodedIRData.address = tValue.UByte.LowByte; // first 8 bit
268  } else {
269  // extended NEC protocol
270  decodedIRData.address = tValue.UWord.LowWord; // first 16 bit
271  }
272  // Check for command if it is 8 bit NEC or 16 bit ONKYO
273  if (tValue.UByte.MidHighByte == (uint8_t) (~tValue.UByte.HighByte)) {
274  decodedIRData.protocol = NEC;
275  } else {
276  decodedIRData.protocol = ONKYO;
277  decodedIRData.command = tValue.UWord.HighWord; // 16 bit command
278 
279  /*
280  * Old NEC plausibility check below, now it is just ONKYO :-)
281  */
282 // IR_DEBUG_PRINT(F("NEC: "));
283 // IR_DEBUG_PRINT(F("Command=0x"));
284 // IR_DEBUG_PRINT(tValue.UByte.MidHighByte, HEX);
285 // IR_DEBUG_PRINT(F(" is not inverted value of 0x"));
286 // IR_DEBUG_PRINTLN(tValue.UByte.HighByte, HEX);
287 // decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
288  }
289  }
290  decodedIRData.numberOfBits = NEC_BITS;
291 
292  // check for special repeat, do not check for same content ;-)
293  if (decodedIRData.rawDataPtr->rawbuf[0] < (NEC_MAXIMUM_REPEAT_SPACE / MICROS_PER_TICK)) {
294  decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_SPECIAL_REPEAT;
295  }
296 
297  return true;
298 }

300 bool IRrecv::decodeNECMSB(decode_results *aResults) {
301  unsigned int offset = 1; // Index in to results; Skip first space.
302 
303 // Check header "mark"
304  if (!matchMark(aResults->rawbuf[offset], NEC_HEADER_MARK)) {
305  return false;
306  }
307  offset++;
308 
309 // Check for repeat
310  if ((aResults->rawlen == 4) && matchSpace(aResults->rawbuf[offset], NEC_REPEAT_HEADER_SPACE)
311  && matchMark(aResults->rawbuf[offset + 1], NEC_BIT_MARK)) {
312  aResults->bits = 0;
313  aResults->value = 0xFFFFFFFF;
314  decodedIRData.flags |= IRDATA_FLAGS_IS_REPEAT;
315  decodedIRData.protocol = NEC;
316  return true;
317  }
318 
319  // Check we have the right amount of data (32). +4 for initial gap, start bit mark and space + stop bit mark
320  if (aResults->rawlen != (2 * NEC_BITS) + 4) {
321  IR_DEBUG_PRINT(F("NEC MSB: "));
322  IR_DEBUG_PRINT(F("Data length="));
323  IR_DEBUG_PRINT(aResults->rawlen);
324  IR_DEBUG_PRINTLN(F(" is not 68"));
325  return false;
326  }
327 
328 // Check header "space"
329  if (!matchSpace(aResults->rawbuf[offset], NEC_HEADER_SPACE)) {
330  IR_DEBUG_PRINT(F("NEC MSB: "));
331  IR_DEBUG_PRINTLN(F("Header space length is wrong"));
332  return false;
333  }
334  offset++;
335 
336  if (!decodePulseDistanceData(NEC_BITS, offset, NEC_BIT_MARK, NEC_ONE_SPACE, NEC_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
337  IR_DEBUG_PRINT(F("NEC MSB: "));
338  IR_DEBUG_PRINTLN(F("Decode failed"));
339  return false;
340  }
341 
342  // Stop bit
343  if (!matchMark(aResults->rawbuf[offset + (2 * NEC_BITS)], NEC_BIT_MARK)) {
344  IR_DEBUG_PRINT(F("NEC MSB: "));
345  IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
346  return false;
347  }
348 
349 // Success
350  aResults->value = decodedIRData.decodedRawData;
351  aResults->bits = NEC_BITS;
352  aResults->decode_type = NEC;
353  decodedIRData.protocol = NEC;
354 
355  return true;
356 }
