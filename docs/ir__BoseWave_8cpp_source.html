18 // BBBB OOO SSSS EEEEE
19 // B B O O S E
20 // BB B O O SSS EEEE
21 // B B O O S E
22 // BBBB OOO SSSS EEEEE

24 // see http://lirc.sourceforge.net/remotes/bose/WAVERADIO
25 //
26 // Support for Bose Wave Radio CD initially provided by https://github.com/uvotguy.
27 //
28 // As seen on my trusty oscilloscope, there is no repeat code. Instead, when I
29 // press and hold a button on my remote, it sends a command, makes a 51.2ms space,
30 // and resends the command, etc, etc.
31 
32 // LSB first, 1 start bit + 8 bit data + 8 bit inverted data + 1 stop bit.
33 #define BOSEWAVE_BITS 16 // Command and inverted command
34 
35 #define BOSEWAVE_HEADER_MARK 1060
36 #define BOSEWAVE_HEADER_SPACE 1450
37 #define BOSEWAVE_BIT_MARK 534
38 #define BOSEWAVE_ONE_SPACE 468
39 #define BOSEWAVE_ZERO_SPACE 1447
40 
41 #define BOSEWAVE_REPEAT_SPACE 52000

45 void IRsend::sendBoseWave(uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
46  // Set IR carrier frequency
47  enableIROut(38);
48 
49  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
50  while (tNumberOfCommands > 0) {
51 
52  // Header
53  mark(BOSEWAVE_HEADER_MARK);
54  space(BOSEWAVE_HEADER_SPACE);
55  // send 8 command bits and then 8 inverted command bits LSB first
56  uint16_t tData = ((~aCommand) << 8) | aCommand;
57 
58  sendPulseDistanceWidthData(BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_BIT_MARK, BOSEWAVE_ZERO_SPACE, tData,
59  BOSEWAVE_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
60 
61  tNumberOfCommands--;
62  // skip last delay!
63  if (tNumberOfCommands > 0) {
64  // send repeated command with a fixed space gap
65  delay( BOSEWAVE_REPEAT_SPACE / 1000);
66  }
67  }
68 }

71 bool IRrecv::decodeBoseWave() {
72 
73  // Check header "mark"
74  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], BOSEWAVE_HEADER_MARK)) {
75  // no debug output, since this check is mainly to determine the received protocol
76  return false;
77  }
78 
79  // Check we have enough data +4 for initial gap, start bit mark and space + stop bit mark
80  if (decodedIRData.rawDataPtr->rawlen != (2 * BOSEWAVE_BITS) + 4) {
81  DEBUG_PRINT("Bose: ");
82  DEBUG_PRINT("Data length=");
83  DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
84  DEBUG_PRINTLN(" is not 36");
85  return false;
86  }
87  // Check header "space"
88  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], BOSEWAVE_HEADER_SPACE)) {
89  DEBUG_PRINT("Bose: ");
90  DEBUG_PRINTLN("Header space length is wrong");
91  return false;
92  }
93 
94  if (!decodePulseDistanceData(BOSEWAVE_BITS, 3, BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
95  DEBUG_PRINT("Bose: ");
96  DEBUG_PRINTLN("Decode failed");
97  return false;
98  }
99 
100  // Stop bit
101  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * BOSEWAVE_BITS)], BOSEWAVE_BIT_MARK)) {
102  DEBUG_PRINT("Bose: ");
103  DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
104  return false;
105  }
106 
107  // Success
108 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
109  uint16_t tDecodedValue = decodedIRData.decodedRawData;
110  uint8_t tCommandNotInverted = tDecodedValue & 0xFF;
111  uint8_t tCommandInverted = tDecodedValue >> 8;
112  // parity check for command. Use this variant to avoid compiler warning "comparison of promoted ~unsigned with unsigned [-Wsign-compare]"
113  if ((tCommandNotInverted ^ tCommandInverted) != 0xFF) {
114  DEBUG_PRINT("Bose: ");
115  DEBUG_PRINT("Command and inverted command check failed");
116  return false;
117  }
118 
119  // check for repeat
120  if (decodedIRData.rawDataPtr->rawbuf[0] < ((BOSEWAVE_REPEAT_SPACE + (BOSEWAVE_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
121  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
122  }
123 
124  decodedIRData.command = tCommandNotInverted;
125  decodedIRData.protocol = BOSEWAVE;
126  decodedIRData.numberOfBits = BOSEWAVE_BITS;
127 
128  return true;
129 }
