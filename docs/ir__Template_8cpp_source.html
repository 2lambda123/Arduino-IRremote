143 // LSB first, 1 start bit + 16 bit address + 8 bit command + 1 stop bit.
144 #define SHUZU_ADDRESS_BITS 16 // 16 bit address
145 #define SHUZU_COMMAND_BITS 8 // Command
146 
147 #define SHUZU_BITS (SHUZU_ADDRESS_BITS + SHUZU_COMMAND_BITS) // The number of bits in the protocol
148 #define SHUZU_UNIT 560
149 
150 #define SHUZU_HEADER_MARK (16 * SHUZU_UNIT) // The length of the Header:Mark
151 #define SHUZU_HEADER_SPACE (8 * SHUZU_UNIT) // The length of the Header:Space
152 
153 #define SHUZU_BIT_MARK SHUZU_UNIT // The length of a Bit:Mark
154 #define SHUZU_ONE_SPACE (3 * SHUZU_UNIT) // The length of a Bit:Space for 1's
155 #define SHUZU_ZERO_SPACE SHUZU_UNIT // The length of a Bit:Space for 0's
156 
157 #define SHUZU_REPEAT_HEADER_SPACE (4 * SHUZU_UNIT) // 2250
158 
159 #define SHUZU_REPEAT_SPACE 45000
160 
161 #define SHUZU_OTHER 1234 // Other things you may need to define

165 void IRsend::sendShuzu(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
166  // Set IR carrier frequency
167  enableIROut(37); // 36.7kHz is the correct frequency
168 
169  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
170  while (tNumberOfCommands > 0) {
171 
172  // Header
173  mark(SHUZU_HEADER_MARK);
174  space(SHUZU_HEADER_SPACE);
175 
176  // Address (device and subdevice)
177  sendPulseDistanceWidthData(SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_BIT_MARK, SHUZU_ZERO_SPACE, aAddress,
178  SHUZU_ADDRESS_BITS, PROTOCOL_IS_LSB_FIRST); // false -> LSB first
179 
180  // Command + stop bit
181  sendPulseDistanceWidthData(SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_BIT_MARK, SHUZU_ZERO_SPACE, aCommand,
182  SHUZU_COMMAND_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT); // false, true -> LSB first, stop bit
183 
184  tNumberOfCommands--;
185  // skip last delay!
186  if (tNumberOfCommands > 0) {
187  // send repeated command in a fixed raster
188  delay(SHUZU_REPEAT_SPACE / 1000);
189  }
190  }
191 }

196  * First check for right data length
197  * Next check start bit
198  * Next try the decode
199  * Last check stop bit
200  */
201 bool IRrecv::decodeShuzu() {
202 
203  // Check we have the right amount of data (28). The +4 is for initial gap, start bit mark and space + stop bit mark
204  if (decodedIRData.rawDataPtr->rawlen != (2 * SHUZU_BITS) + 4) {
205  // no debug output, since this check is mainly to determine the received protocol
206  return false;
207  }
208 
209  // Check header "space"
210  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], SHUZU_HEADER_MARK) || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], SHUZU_HEADER_SPACE)) {
211  DBG_PRINT("Shuzu: ");
212  DBG_PRINTLN("Header mark or space length is wrong");
213  return false;
214  }
215 
216  // false -> LSB first
217  if (!decodePulseDistanceData(SHUZU_BITS, 3, SHUZU_BIT_MARK, SHUZU_ONE_SPACE, SHUZU_ZERO_SPACE, false)) {
218  DBG_PRINT(F("Shuzu: "));
219  DBG_PRINTLN(F("Decode failed"));
220  return false;
221  }
222 
223  // Stop bit
224  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * SHUZU_BITS)], SHUZU_BIT_MARK)) {
225  DBG_PRINT(F("Shuzu: "));
226  DBG_PRINTLN(F("Stop bit mark length is wrong"));
227  return false;
228  }
229 
230  // Success
231 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
232  uint8_t tCommand = decodedIRData.decodedRawData >> SHUZU_ADDRESS_BITS; // upper 8 bits of LSB first value
233  uint8_t tAddress = decodedIRData.decodedRawData & 0xFFFF; // lowest 16 bit of LSB first value
234 
235  /*
236  * Check for repeat
237  */
238  if (decodedIRData.rawDataPtr->rawbuf[0] < ((SHUZU_REPEAT_SPACE + (SHUZU_REPEAT_SPACE / 2)) / MICROS_PER_TICK)) {
239  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
240  }
241  decodedIRData.command = tCommand;
242  decodedIRData.address = tAddress;
243  decodedIRData.numberOfBits = SHUZU_BITS;
244  decodedIRData.protocol = LG; // we have no SHUZU code
245 
246  return true;
247 }
