42 /*
43  * FAST_8_BIT_CS Protocol characteristics:
44  * - Bit timing is like JVC
45  * - The header is shorter, 4000 vs. 12500
46  * - No address and 16 bit data, interpreted as 8 bit command and 8 bit inverted command,
47  * leading to a fixed protocol length of (7 + (16 * 2) + 1) * 526 = 40 * 560 = 21040 microseconds or 21 ms.
48  * - Repeats are sent as complete frames but in a 50 ms period.
49  */

60 void sendMark(uint8_t aSendPin, unsigned int aMarkMicros) {
61  unsigned long tStartMicros = micros();
62  unsigned long tNextPeriodEnding = tStartMicros;
63  unsigned long tMicros;
64  do {
65  /*
66  * Generate pulse
67  */
68  noInterrupts(); // do not let interrupts extend the short on period
69  digitalWriteFast(aSendPin, HIGH);
70  delayMicroseconds(8); // 8 us for a 30 % duty cycle for 38 kHz
71  digitalWriteFast(aSendPin, LOW);
72  interrupts(); // Enable interrupts - to keep micros correct- for the longer off period 3.4 us until receive ISR is active (for 7 us + pop's)
73 
74  /*
75  * PWM pause timing and end check
76  * Minimal pause duration is 4.3 us
77  */
78  tNextPeriodEnding += 26; // for 38 kHz
79  do {
80  tMicros = micros(); // we have only 4 us resolution for AVR @16MHz
81  /*
82  * Exit the forever loop if aMarkMicros has reached
83  */
84  unsigned int tDeltaMicros = tMicros - tStartMicros;
85 #if defined(__AVR__)
86  // Just getting variables and check for end condition takes minimal 3.8 us
87  if (tDeltaMicros >= aMarkMicros - (112 / (F_CPU / MICROS_IN_ONE_SECOND))) { // To compensate for call duration - 112 is an empirical value
88 #else
89  if (tDeltaMicros >= aMarkMicros) {
90  #endif
91  return;
92  }
93  } while (tMicros < tNextPeriodEnding);
94  } while (true);
95 }

97 void sendNECMinimal(uint8_t aSendPin, uint8_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
98  pinModeFast(aSendPin, OUTPUT);
99 
100  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
101  while (tNumberOfCommands > 0) {
102  unsigned long tStartOfFrameMillis = millis();
103 
104  sendMark(aSendPin, NEC_HEADER_MARK);
105  if (tNumberOfCommands < aNumberOfRepeats + 1) {
106  // send the NEC special repeat
107  delayMicroseconds(NEC_REPEAT_HEADER_SPACE); // - 2250
108  } else {
109  // send header
110  delayMicroseconds(NEC_HEADER_SPACE);
111  LongUnion tData;
112  tData.UByte.LowByte = aAddress; // LSB first
113  tData.UByte.MidLowByte = ~aAddress;
114  tData.UByte.MidHighByte = aCommand;
115  tData.UByte.HighByte = ~aCommand; // LSB first
116  // Send data
117  for (uint_fast8_t i = 0; i < NEC_BITS; ++i) {
118  sendMark(aSendPin, NEC_BIT_MARK); // constant mark length
119 
120  if (tData.ULong & 1) {
121  delayMicroseconds(NEC_ONE_SPACE);
122  } else {
123  delayMicroseconds(NEC_ZERO_SPACE);
124  }
125  tData.ULong >>= 1; // shift command for next bit
126  }
127  } // send stop bit
128  sendMark(aSendPin, NEC_BIT_MARK);
129 
130  tNumberOfCommands--;
131  // skip last delay!
132  if (tNumberOfCommands > 0) {
133  /*
134  * Check and fallback for wrong RepeatPeriodMillis parameter. I.e the repeat period must be greater than each frame duration.
135  */
136  auto tFrameDurationMillis = millis() - tStartOfFrameMillis;
137  if (NEC_REPEAT_PERIOD / 1000 > tFrameDurationMillis) {
138  delay(NEC_REPEAT_PERIOD / 1000 - tFrameDurationMillis);
139  }
140  }
141  }
142 }

147 void sendFast8BitAndParity(uint8_t aSendPin, uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
148  pinModeFast(aSendPin, OUTPUT);
149 
150  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
151  while (tNumberOfCommands > 0) {
152  unsigned long tStartOfFrameMillis = millis();
153 
154  // send header
155  sendMark(aSendPin, FAST_8_BIT_PARITY_HEADER_MARK);
156  delayMicroseconds(FAST_8_BIT_PARITY_HEADER_SPACE);
157  uint16_t tData = aCommand | (((uint8_t) (~aCommand)) << 8); // LSB first
158  // Send data
159  for (uint_fast8_t i = 0; i < FAST_8_BIT_PARITY_BITS; ++i) {
160  sendMark(aSendPin, FAST_8_BIT_PARITY_BIT_MARK); // constant mark length
161 
162  if (tData & 1) {
163  delayMicroseconds(FAST_8_BIT_PARITY_ONE_SPACE);
164  } else {
165  delayMicroseconds(FAST_8_BIT_PARITY_ZERO_SPACE);
166  }
167  tData >>= 1; // shift command for next bit
168  }
169  // send stop bit
170  sendMark(aSendPin, FAST_8_BIT_PARITY_BIT_MARK);
171 
172  tNumberOfCommands--;
173  // skip last delay!
174  if (tNumberOfCommands > 0) {
175  /*
176  * Check and fallback for wrong RepeatPeriodMillis parameter. I.e the repeat period must be greater than each frame duration.
177  */
178  auto tFrameDurationMillis = millis() - tStartOfFrameMillis;
