48 // see http://www.hifi-remote.com/wiki/index.php?title=DecodeIR#Samsung
49 // https://www.mikrocontroller.net/articles/IRMP_-_english#SAMSUNG32
50 // LSB first, 1 start bit + 16 bit address + 16,32,20 bit data + 1 stop bit.
51 // repeats are like NEC but with 2 stop bits
52 
53 #define SAMSUNG_ADDRESS_BITS 16
54 #define SAMSUNG_COMMAND16_BITS 16
55 #define SAMSUNG_COMMAND32_BITS 32
56 #define SAMSUNG_BITS (SAMSUNG_ADDRESS_BITS + SAMSUNG_COMMAND16_BITS)
57 #define SAMSUNG48_BITS (SAMSUNG_ADDRESS_BITS + SAMSUNG_COMMAND32_BITS)
58 
59 #define SAMSUNG_UNIT 550
60 #define SAMSUNG_HEADER_MARK (8 * SAMSUNG_UNIT) // 4400
61 #define SAMSUNG_HEADER_SPACE (8 * SAMSUNG_UNIT) // 4400
62 #define SAMSUNG_BIT_MARK SAMSUNG_UNIT
63 #define SAMSUNG_ONE_SPACE (3 * SAMSUNG_UNIT) // 1650
64 #define SAMSUNG_ZERO_SPACE SAMSUNG_UNIT
65 
66 #define SAMSUNG_AVERAGE_DURATION 55000 // SAMSUNG_HEADER_MARK + SAMSUNG_HEADER_SPACE + 32 * 2,5 * SAMSUNG_UNIT + SAMSUNG_UNIT // 2.5 because we assume more zeros than ones
67 #define SAMSUNG_REPEAT_DURATION (SAMSUNG_HEADER_MARK + SAMSUNG_HEADER_SPACE + SAMSUNG_BIT_MARK + SAMSUNG_ZERO_SPACE + SAMSUNG_BIT_MARK)
68 #define SAMSUNG_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.

84 void IRsend::sendSamsung(uint16_t aAddress, uint16_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
85  if (aIsRepeat) {
86  sendSamsungRepeat();
87  return;
88  }
89 
90  // Set IR carrier frequency
91  enableIROut(38);
92 
93  // Header
94  mark(SAMSUNG_HEADER_MARK);
95  space(SAMSUNG_HEADER_SPACE);
96 
97  // Address
98  sendPulseDistanceWidthData(SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_BIT_MARK, SAMSUNG_ZERO_SPACE, aAddress,
99  SAMSUNG_ADDRESS_BITS, PROTOCOL_IS_LSB_FIRST);
100 
101  // Command
102 
103  // send 8 command bits and then 8 inverted command bits LSB first
104  aCommand = aCommand & 0xFF;
105  aCommand = ((~aCommand) << 8) | aCommand;
106 
107  sendPulseDistanceWidthData(SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_BIT_MARK, SAMSUNG_ZERO_SPACE, aCommand,
108  SAMSUNG_COMMAND16_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
109 
110  for (uint_fast8_t i = 0; i < aNumberOfRepeats; ++i) {
111  // send repeat in a 110 ms raster
112  if (i == 0) {
113  delay((SAMSUNG_REPEAT_PERIOD - SAMSUNG_AVERAGE_DURATION) / 1000);
114  } else {
115  delay((SAMSUNG_REPEAT_PERIOD - SAMSUNG_REPEAT_DURATION) / 1000);
116  }
117  // send repeat
118  sendSamsungRepeat();
119  }
120 }

123 bool IRrecv::decodeSamsung() {
124 
125  // Check we have enough data (68). The +4 is for initial gap, start bit mark and space + stop bit mark
126  if (decodedIRData.rawDataPtr->rawlen != ((2 * SAMSUNG_BITS) + 4)
127  && decodedIRData.rawDataPtr->rawlen != ((2 * SAMSUNG48_BITS) + 4) && (decodedIRData.rawDataPtr->rawlen != 6)) {
128  DEBUG_PRINT(F("Samsung: "));
129  DEBUG_PRINT("Data length=");
130  DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
131  DEBUG_PRINTLN(" is not 68 or 100 or 6");
132  return false;
133  }
134 
135  // Check header "mark" + "space"
136  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], SAMSUNG_HEADER_MARK)
137  || !matchSpace(decodedIRData.rawDataPtr->rawbuf[2], SAMSUNG_HEADER_SPACE)) {
138  DEBUG_PRINT("Samsung: ");
139  DEBUG_PRINTLN("Header mark or space length is wrong");
140 
141  return false;
142  }
143 
144  // Check for repeat
145  if (decodedIRData.rawDataPtr->rawlen == 6) {
146  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT| IRDATA_FLAGS_IS_LSB_FIRST;
147  decodedIRData.address = lastDecodedAddress;
148  decodedIRData.command = lastDecodedCommand;
149  decodedIRData.protocol = SAMSUNG;
150  return true;
151  }
152 
153  if (decodedIRData.rawDataPtr->rawlen == (2 * SAMSUNG48_BITS) + 4) {
154  /*
155  * Samsung48
156  */
157  // decode address
158  if (!decodePulseDistanceData(SAMSUNG_ADDRESS_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
159  DEBUG_PRINT("Samsung: ");
160  DEBUG_PRINTLN("Decode failed");
161  return false;
162  }
163  decodedIRData.address = decodedIRData.decodedRawData;
164 
165  // decode 32 bit command
166  if (!decodePulseDistanceData(SAMSUNG_COMMAND32_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
167  DEBUG_PRINT("Samsung: ");
168  DEBUG_PRINTLN("Decode failed");
169  return false;
170  }
171 
172  // Success
173 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
174  LongUnion tValue;
175  tValue.ULong = decodedIRData.decodedRawData;
176  // receive 2 * (8 bits then 8 inverted bits) LSB first
177  if (tValue.UByte.HighByte != (uint8_t) (~tValue.UByte.MidHighByte)
178  && tValue.UByte.MidLowByte != (uint8_t) (~tValue.UByte.LowByte)) {
179  decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
180  }
181  decodedIRData.command = tValue.UByte.HighByte << 8 | tValue.UByte.MidLowByte;
182  decodedIRData.numberOfBits = SAMSUNG48_BITS;
183 
184  } else {
185  /*
186  * Samsung32
187  */
188  if (!decodePulseDistanceData(SAMSUNG_BITS, 3, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_LSB_FIRST)) {
189  DEBUG_PRINT("Samsung: ");
190  DEBUG_PRINTLN("Decode failed");
191  return false;
192  }
193  LongUnion tValue;
194  tValue.ULong = decodedIRData.decodedRawData;
195  decodedIRData.address = tValue.UWord.LowWord;
196 
197  if (tValue.UByte.MidHighByte == (uint8_t) (~tValue.UByte.HighByte)) {
198  // 8 bit command protocol
199  decodedIRData.command = tValue.UByte.MidHighByte; // first 8 bit
200  } else {
201  // 16 bit command protocol
202  decodedIRData.command = tValue.UWord.HighWord; // first 16 bit
203  }
204  decodedIRData.numberOfBits = SAMSUNG_BITS;
205  }
206 
207  decodedIRData.protocol = SAMSUNG;
208 
209  return true;
210 }

213 bool IRrecv::decodeSAMSUNG(decode_results *aResults) {
214  unsigned int offset = 1; // Skip first space
215 
216  // Initial mark
217  if (!matchMark(aResults->rawbuf[offset], SAMSUNG_HEADER_MARK)) {
218  return false;
219  }
220  offset++;
221 
222 // Check for repeat -- like a NEC repeat
223  if ((aResults->rawlen == 4) && matchSpace(aResults->rawbuf[offset], 2250)
224  && matchMark(aResults->rawbuf[offset + 1], SAMSUNG_BIT_MARK)) {
225  aResults->bits = 0;
226  aResults->value = 0xFFFFFFFF;
227  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
228  decodedIRData.protocol = SAMSUNG;
229  return true;
230  }
231  if (aResults->rawlen < (2 * SAMSUNG_BITS) + 4) {
232  return false;
233  }
234 
235 // Initial space
236  if (!matchSpace(aResults->rawbuf[offset], SAMSUNG_HEADER_SPACE)) {
237  return false;
238  }
239  offset++;
240 
241  if (!decodePulseDistanceData(SAMSUNG_BITS, offset, SAMSUNG_BIT_MARK, SAMSUNG_ONE_SPACE, SAMSUNG_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
242  return false;
243  }
244 
245 // Success
246  aResults->value = decodedIRData.decodedRawData;
247  aResults->bits = SAMSUNG_BITS;
248  aResults->decode_type = SAMSUNG;
249  decodedIRData.protocol = SAMSUNG;
250  return true;
251 }
