39 //==============================================================================
40 // N N EEEEE CCCC
41 // NN N E C
42 // N N N EEE C
43 // N NN E C
44 // N N EEEEE CCCC
45 //==============================================================================
46 // see: https://www.sbprojects.net/knowledge/ir/nec.php
47 
48 // for Apple see https://en.wikipedia.org/wiki/Apple_Remote
49 
50 // LSB first, 1 start bit + 16 bit address + 8 bit command + 8 bit inverted command + 1 stop bit.
51 //
52 #define NEC_ADDRESS_BITS 16 // 16 bit address or 8 bit address and 8 bit inverted address
53 #define NEC_COMMAND_BITS 16 // Command and inverted command
54 
55 #define NEC_BITS (NEC_ADDRESS_BITS + NEC_COMMAND_BITS)
56 #define NEC_UNIT 560
57 
58 #define NEC_HEADER_MARK (16 * NEC_UNIT) // 9000
59 #define NEC_HEADER_SPACE (8 * NEC_UNIT) // 4500
60 
61 #define NEC_BIT_MARK NEC_UNIT
62 #define NEC_ONE_SPACE (3 * NEC_UNIT) // 1690
63 #define NEC_ZERO_SPACE NEC_UNIT
64 
65 #define NEC_REPEAT_HEADER_SPACE (4 * NEC_UNIT) // 2250
66 
67 #define NEC_AVERAGE_DURATION 62000 // NEC_HEADER_MARK + NEC_HEADER_SPACE + 32 * 2,5 * NEC_UNIT + NEC_UNIT // 2.5 because we assume more zeros than ones
68 #define NEC_REPEAT_DURATION (NEC_HEADER_MARK + NEC_REPEAT_HEADER_SPACE + NEC_BIT_MARK) // 12 ms
69 #define NEC_REPEAT_PERIOD 110000 // Commands are repeated every 110 ms (measured from start to start) for as long as the key on the remote control is held down.
70 #define NEC_REPEAT_SPACE (NEC_REPEAT_PERIOD - NEC_AVERAGE_DURATION) // 48 ms
71 
72 #define APPLE_ADDRESS 0x87EE

86 /*
87  * Repeat commands should be sent in a 110 ms raster.
88  * There is NO delay after the last sent repeat!
89  * https://www.sbprojects.net/knowledge/ir/nec.php
90  * @param aIsRepeat if true, send only one repeat frame without leading and trailing space
91  */
92 void IRsend::sendNEC(uint16_t aAddress, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
93 
94  LongUnion tRawData;
95 
96  // Address 16 bit LSB first
97  if ((aAddress & 0xFF00) == 0) {
98  // assume 8 bit address -> send 8 address bits and then 8 inverted address bits LSB first
99  tRawData.UByte.LowByte = aAddress;
100  tRawData.UByte.MidLowByte = ~tRawData.UByte.LowByte;
101  } else {
102  tRawData.UWord.LowWord = aAddress;
103  }
104 
105  // send 8 command bits and then 8 inverted command bits LSB first
106  tRawData.UByte.MidHighByte = aCommand;
107  tRawData.UByte.HighByte = ~aCommand;
108 
109  sendNECRaw(tRawData.ULong, aNumberOfRepeats, aIsRepeat);
110 }

112 /*
113  * Repeat commands should be sent in a 110 ms raster.
114  * There is NO delay after the last sent repeat!
115  * https://en.wikipedia.org/wiki/Apple_Remote
116  * https://gist.github.com/darconeous/4437f79a34e3b6441628
117  * @param aAddress is the DeviceId*
118  * @param aIsRepeat if true, send only one repeat frame without leading and trailing space
119  */
120 void IRsend::sendApple(uint8_t aDeviceId, uint8_t aCommand, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
121 
122  LongUnion tRawData;
123 
124  // Address 16 bit LSB first
125  tRawData.UWord.LowWord = APPLE_ADDRESS;
126 
127  // send Apple code and then 8 command bits LSB first
128  tRawData.UByte.MidHighByte = aCommand;
129  tRawData.UByte.HighByte = aDeviceId; // e.g. 0xD7
130 
131  sendNECRaw(tRawData.ULong, aNumberOfRepeats, aIsRepeat);
132 }

134 void IRsend::sendNECRaw(uint32_t aRawData, uint_fast8_t aNumberOfRepeats, bool aIsRepeat) {
135  if (aIsRepeat) {
136  sendNECRepeat();
137  return;
138  }
139  // Set IR carrier frequency
140  enableIROut(38);
141 
142  // Header
143  mark(NEC_HEADER_MARK);
144  space(NEC_HEADER_SPACE);
145 
146  // LSB first + stop bit
147  sendPulseDistanceWidthData(NEC_BIT_MARK, NEC_ONE_SPACE, NEC_BIT_MARK, NEC_ZERO_SPACE, aRawData, NEC_BITS, PROTOCOL_IS_LSB_FIRST,
148  SEND_STOP_BIT);
149 
150  for (uint_fast8_t i = 0; i < aNumberOfRepeats; ++i) {
151  // send repeat in a 110 ms raster
152  if (i == 0) {
153  delay(NEC_REPEAT_SPACE / 1000);
154  } else {
155  delay((NEC_REPEAT_PERIOD - NEC_REPEAT_DURATION) / 1000);
156  }
157  // send repeat
158  sendNECRepeat();
159  }
160 }

174 bool IRrecv::decodeNEC() {
175 
176  // Check we have the right amount of data (68). The +4 is for initial gap, start bit mark and space + stop bit mark.
177  if (decodedIRData.rawDataPtr->rawlen != ((2 * NEC_BITS) + 4) && (decodedIRData.rawDataPtr->rawlen != 4)) {
178  // no debug output, since this check is mainly to determine the received protocol
179  return false;
180  }
181 
182  // Check header "mark" this must be done for repeat and data
183  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], NEC_HEADER_MARK)) {
184  return false;
185  }
186 
187  // Check for repeat - here we have another header space length
188  if (decodedIRData.rawDataPtr->rawlen == 4) {
189  if (matchSpace(decodedIRData.rawDataPtr->rawbuf[2], NEC_REPEAT_HEADER_SPACE)
190  && matchMark(decodedIRData.rawDataPtr->rawbuf[3], NEC_BIT_MARK)) {
191  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
192  decodedIRData.address = lastDecodedAddress;
193  decodedIRData.command = lastDecodedCommand;
194 // decodedIRData.protocol = NEC; do not set it, because it can also be an LG repeat
195  return true;
196  }
197  return false;
198  }
199 
200  // Check command header space
201  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], NEC_HEADER_SPACE)) {
202  DBG_PRINT(F("NEC: "));
203  DBG_PRINTLN(F("Header space length is wrong"));
204  return false;
205  }
206 
207  if (!decodePulseDistanceData(NEC_BITS, 3, NEC_BIT_MARK, NEC_ONE_SPACE, NEC_ZERO_SPACE, false)) {
208  DBG_PRINT(F("NEC: "));
209  DBG_PRINTLN(F("Decode failed"));
210  return false;
211  }
212 
213  // Stop bit
214  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * NEC_BITS)], NEC_BIT_MARK)) {
215  DBG_PRINT(F("NEC: "));
216  DBG_PRINTLN(F("Stop bit mark length is wrong"));
217  return false;
218  }
219 
220  // Success
221 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
222  LongUnion tValue;
223  tValue.ULong = decodedIRData.decodedRawData;
224  decodedIRData.command = tValue.UByte.MidHighByte;
225  // Address
226  if (tValue.UWord.LowWord == APPLE_ADDRESS) {
227  /*
228  * Apple
229  */
230  decodedIRData.protocol = APPLE;
231  decodedIRData.command = tValue.UByte.MidHighByte;
232  decodedIRData.address = tValue.UByte.HighByte;
233 
234  } else {
235  /*
236  * NEC
237  */
238  decodedIRData.protocol = NEC;
239  if (tValue.UByte.LowByte == (uint8_t) (~tValue.UByte.MidLowByte)) {
240  // standard 8 bit address NEC protocol
241  decodedIRData.address = tValue.UByte.LowByte; // first 8 bit
242  } else {
243  // extended NEC protocol
244  decodedIRData.address = tValue.UWord.LowWord; // first 16 bit
245  }
246  // plausi check for command
247  if (tValue.UByte.MidHighByte != (uint8_t) (~tValue.UByte.HighByte)) {
248  DBG_PRINT(F("NEC: "));
249  DBG_PRINT(F("Command=0x"));
250  DBG_PRINT(tValue.UByte.MidHighByte, HEX);
251  DBG_PRINT(F(" is not inverted value of 0x"));
252  DBG_PRINTLN(tValue.UByte.HighByte, HEX);
253  decodedIRData.flags = IRDATA_FLAGS_PARITY_FAILED | IRDATA_FLAGS_IS_LSB_FIRST;
254  }
255  }
256  decodedIRData.numberOfBits = NEC_BITS;
257 
258  return true;
259 }

262 bool IRrecv::decodeNEC() {
263  unsigned int offset = 1; // Index in to results; Skip first space.
264 
265 // Check header "mark"
266  if (!matchMark(results.rawbuf[offset], NEC_HEADER_MARK)) {
267  return false;
268  }
269  offset++;
270 
271 // Check for repeat
272  if ((results.rawlen == 4) && matchSpace(results.rawbuf[offset], NEC_REPEAT_HEADER_SPACE)
273  && matchMark(results.rawbuf[offset + 1], NEC_BIT_MARK)) {
274  results.bits = 0;
275  results.value = 0xFFFFFFFF;
276  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT;
277  decodedIRData.protocol = NEC;
278  return true;
279  }
280 
281  // Check we have the right amount of data (32). +4 for initial gap, start bit mark and space + stop bit mark
282  if (results.rawlen != (2 * NEC_BITS) + 4) {
283  DBG_PRINT("NEC: ");
284  DBG_PRINT("Data length=");
285  DBG_PRINT(results.rawlen);
286  DBG_PRINTLN(" is not 68");
287  return false;
288  }
289 
290 // Check header "space"
291  if (!matchSpace(results.rawbuf[offset], NEC_HEADER_SPACE)) {
292  DBG_PRINT("NEC: ");
293  DBG_PRINTLN("Header space length is wrong");
294  return false;
295  }
296  offset++;
297 
298  if (!decodePulseDistanceData(NEC_BITS, offset, NEC_BIT_MARK, NEC_ONE_SPACE, NEC_ZERO_SPACE, PROTOCOL_IS_MSB_FIRST)) {
299  return false;
300  }
301 
302  // Stop bit
303  if (!matchMark(results.rawbuf[offset + (2 * NEC_BITS)], NEC_BIT_MARK)) {
304  DBG_PRINT("NEC: ");
305  DBG_PRINTLN(F("Stop bit mark length is wrong"));
306  return false;
307  }
308 
309 // Success
310  results.bits = NEC_BITS;
311  decodedIRData.protocol = NEC;
312 
313  return true;
314 }

317 /*
318  * With Send sendNECMSB() you can send your old 32 bit codes.
319  * To convert one into the other, you must reverse the byte positions and then reverse all bit positions of each byte.
320  * Or write it as one binary string and reverse/mirror it.
321  * Example:
322  * 0xCB340102 byte reverse -> 02 01 34 CB bit reverse-> 40 80 2C D3.
323  * 0xCB340102 is binary 11001011001101000000000100000010.
324  * 0x40802CD3 is binary 01000000100000000010110011010011.
325  * If you read the first binary sequence backwards (right to left), you get the second sequence.
326  */
327 void IRsend::sendNECMSB(uint32_t data, uint8_t nbits, bool repeat) {
328  // Set IR carrier frequency
329  enableIROut(38);
330 
331  if (data == 0xFFFFFFFF || repeat) {
332  sendNECRepeat();
333  return;
334  }
335 
336  // Header
337  mark(NEC_HEADER_MARK);
338  space(NEC_HEADER_SPACE);
339 
340  // Old version with MSB first Data + stop bit
341  sendPulseDistanceWidthData(NEC_BIT_MARK, NEC_ONE_SPACE, NEC_BIT_MARK, NEC_ZERO_SPACE, data, nbits, PROTOCOL_IS_MSB_FIRST,
342  SEND_STOP_BIT);
343 }
