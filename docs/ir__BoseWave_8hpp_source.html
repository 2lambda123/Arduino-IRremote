21 // BBBB OOO SSSS EEEEE
22 // B B O O S E
23 // BB B O O SSS EEEE
24 // B B O O S E
25 // BBBB OOO SSSS EEEEE

27 // see http://lirc.sourceforge.net/remotes/bose/WAVERADIO
28 // see: https://www.mikrocontroller.net/articles/IRMP_-_english#BOSE
29 //
30 // Support for Bose Wave Radio CD initially provided by https://github.com/uvotguy.
31 //
32 // As seen on my trusty oscilloscope, there is no repeat code. Instead, when I
33 // press and hold a button on my remote, it sends a command, makes a 51.2ms space,
34 // and resends the command, etc, etc.
35 // LSB first, 1 start bit + 8 bit data + 8 bit inverted data + 1 stop bit.
36 #define BOSEWAVE_BITS 16 // Command and inverted command
37 
38 #define BOSEWAVE_HEADER_MARK 1014 // 1014 are 39 clock periods (I counted 3 times!)
39 #define BOSEWAVE_HEADER_SPACE 1468 // 1468(measured), 1456 are 56 clock periods
40 #define BOSEWAVE_BIT_MARK 520 // 520 are 20 clock periods
41 #define BOSEWAVE_ZERO_SPACE 468 // 468 are 18 clock periods
42 #define BOSEWAVE_ONE_SPACE 1468 // 1468(measured), 1456 are 56 clock periods
43 
44 #define BOSEWAVE_REPEAT_SPACE 50000

48 void IRsend::sendBoseWave(uint8_t aCommand, uint_fast8_t aNumberOfRepeats) {
49  // Set IR carrier frequency
50  enableIROut(BOSEWAVE_KHZ); // 38 kHz
51 
52  uint_fast8_t tNumberOfCommands = aNumberOfRepeats + 1;
53  while (tNumberOfCommands > 0) {
54 
55  // Header
56  mark(BOSEWAVE_HEADER_MARK);
57  space(BOSEWAVE_HEADER_SPACE);
58  // send 8 command bits and then 8 inverted command bits LSB first
59  uint16_t tData = ((~aCommand) << 8) | aCommand;
60 
61  sendPulseDistanceWidthData(BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_BIT_MARK, BOSEWAVE_ZERO_SPACE, tData,
62  BOSEWAVE_BITS, PROTOCOL_IS_LSB_FIRST, SEND_STOP_BIT);
63 
64  tNumberOfCommands--;
65  // skip last delay!
66  if (tNumberOfCommands > 0) {
67  // send repeated command with a fixed space gap
68  delay( BOSEWAVE_REPEAT_SPACE / MICROS_IN_ONE_MILLI);
69  }
70  }
71  IrReceiver.restartAfterSend();
72 }

75 bool IRrecv::decodeBoseWave() {
76 
77  // Check header "mark"
78  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[1], BOSEWAVE_HEADER_MARK)) {
79  // no debug output, since this check is mainly to determine the received protocol
80  return false;
81  }
82 
83  // Check we have enough data +4 for initial gap, start bit mark and space + stop bit mark
84  if (decodedIRData.rawDataPtr->rawlen != (2 * BOSEWAVE_BITS) + 4) {
85  IR_DEBUG_PRINT(F("Bose: "));
86  IR_DEBUG_PRINT(F("Data length="));
87  IR_DEBUG_PRINT(decodedIRData.rawDataPtr->rawlen);
88  IR_DEBUG_PRINTLN(F(" is not 36"));
89  return false;
90  }
91  // Check header "space"
92  if (!matchSpace(decodedIRData.rawDataPtr->rawbuf[2], BOSEWAVE_HEADER_SPACE)) {
93  IR_DEBUG_PRINT(F("Bose: "));
94  IR_DEBUG_PRINTLN(F("Header space length is wrong"));
95  return false;
96  }
97 
98  if (!decodePulseDistanceData(BOSEWAVE_BITS, 3, BOSEWAVE_BIT_MARK, BOSEWAVE_ONE_SPACE, BOSEWAVE_ZERO_SPACE,
99  PROTOCOL_IS_LSB_FIRST)) {
100  IR_DEBUG_PRINT(F("Bose: "));
101  IR_DEBUG_PRINTLN(F("Decode failed"));
102  return false;
103  }
104 
105  // Stop bit
106  if (!matchMark(decodedIRData.rawDataPtr->rawbuf[3 + (2 * BOSEWAVE_BITS)], BOSEWAVE_BIT_MARK)) {
107  IR_DEBUG_PRINT(F("Bose: "));
108  IR_DEBUG_PRINTLN(F("Stop bit mark length is wrong"));
109  return false;
110  }
111 
112  // Success
113 // decodedIRData.flags = IRDATA_FLAGS_IS_LSB_FIRST; // Not required, since this is the start value
114  uint16_t tDecodedValue = decodedIRData.decodedRawData;
115  uint8_t tCommandNotInverted = tDecodedValue & 0xFF; // comes first and is in the lower bits (LSB first :-))
116  uint8_t tCommandInverted = tDecodedValue >> 8;
117  // parity check for command. Use this variant to avoid compiler warning "comparison of promoted ~unsigned with unsigned [-Wsign-compare]"
118  if ((tCommandNotInverted ^ tCommandInverted) != 0xFF) {
119  IR_DEBUG_PRINT(F("Bose: "));
120  IR_DEBUG_PRINT(F("Command and inverted command check failed"));
121  return false;
122  }
123 
124  // check for repeat
125  if (decodedIRData.rawDataPtr->rawbuf[0] < ((BOSEWAVE_REPEAT_SPACE + (BOSEWAVE_REPEAT_SPACE / 4)) / MICROS_PER_TICK)) {
126  decodedIRData.flags = IRDATA_FLAGS_IS_REPEAT | IRDATA_FLAGS_IS_LSB_FIRST;
127  }
128 
129  decodedIRData.command = tCommandNotInverted;
130  decodedIRData.protocol = BOSEWAVE;
131  decodedIRData.numberOfBits = BOSEWAVE_BITS;
132 
133  return true;
134 }
