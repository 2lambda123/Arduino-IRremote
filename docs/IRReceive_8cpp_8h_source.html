400  uint16_t aBitSpaceMicros, bool aMSBfirst) {
401 
402  uint16_t *tRawBufPointer = &decodedIRData.rawDataPtr->rawbuf[aStartOffset];
403  uint32_t tDecodedData = 0;
404 
405  if (aMSBfirst) {
406  for (uint_fast8_t i = 0; i < aNumberOfBits; i++) {
407  // Check for variable length mark indicating a 0 or 1
408  if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
409  tDecodedData = (tDecodedData << 1) | 1;
410  TRACE_PRINT('1');
411  } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
412  tDecodedData = (tDecodedData << 1) | 0;
413  TRACE_PRINT('0');
414  } else {
415  DEBUG_PRINT(F("Mark="));
416  DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
417  DEBUG_PRINT(F(" is not "));
418  DEBUG_PRINT(aOneMarkMicros);
419  DEBUG_PRINT(F(" or "));
420  DEBUG_PRINT(aZeroMarkMicros);
421  DEBUG_PRINT(' ');
422  return false;
423  }
424  tRawBufPointer++;
425 
426  if (tRawBufPointer < &decodedIRData.rawDataPtr->rawbuf[decodedIRData.rawDataPtr->rawlen]) {
427  // Assume that last space, which is not recorded, is correct, since we can not check it
428  // Check for constant length space
429  if (!matchSpace(*tRawBufPointer, aBitSpaceMicros)) {
430  DEBUG_PRINT(F("Space="));
431  DEBUG_PRINT(*tRawBufPointer * MICROS_PER_TICK);
432  DEBUG_PRINT(F(" is not "));
433  DEBUG_PRINT(aBitSpaceMicros);
434  DEBUG_PRINT(' ');
435  return false;
436  }
437  tRawBufPointer++;
438  }
439  }
440  TRACE_PRINTLN("");
441  } else {
442  for (uint32_t tMask = 1UL; aNumberOfBits > 0; tMask <<= 1, aNumberOfBits--) {
443 
444  // Check for variable length mark indicating a 0 or 1
445  if (matchMark(*tRawBufPointer, aOneMarkMicros)) {
446  tDecodedData |= tMask; // set the bit
447  TRACE_PRINT('1');
448  } else if (matchMark(*tRawBufPointer, aZeroMarkMicros)) {
449  // do not set the bit

451  } else {
452  DEBUG_PRINT(F("Mark="));

459  return false;
460  }
461  tRawBufPointer++;
462

464  // Assume that last space, which is not recorded, is correct, since we can not check it

603 uint8_t IRrecv::getBiphaselevel() {
604  uint8_t tLevelOfCurrentInterval; // 0 (SPACE) or 1 (MARK)
605 
606  if (sBiphaseDecodeRawbuffOffset >= decodedIRData.rawDataPtr->rawlen) {
607  return SPACE; // After end of recorded buffer, assume space.
608  }
609 
610  tLevelOfCurrentInterval = (sBiphaseDecodeRawbuffOffset) & 1; // on odd rawbuf offsets we have mark timings
611 
612  /*
613  * Setup data if sUsedTimingIntervals is 0
614  */
615  if (sUsedTimingIntervals == 0) {
616  uint16_t tCurrentTimingWith = decodedIRData.rawDataPtr->rawbuf[sBiphaseDecodeRawbuffOffset];
617  uint16_t tMarkExcessCorrection = (tLevelOfCurrentInterval == MARK) ? MARK_EXCESS_MICROS : -MARK_EXCESS_MICROS;
618 
619  if (matchTicks(tCurrentTimingWith, (sBiphaseTimeUnit) + tMarkExcessCorrection)) {
620  sCurrentTimingIntervals = 1;
621  } else if (matchTicks(tCurrentTimingWith, (2 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
622  sCurrentTimingIntervals = 2;
623  } else if (matchTicks(tCurrentTimingWith, (3 * sBiphaseTimeUnit) + tMarkExcessCorrection)) {
624  sCurrentTimingIntervals = 3;
625  } else {
626  return -1;
627  }
628  }
629 
630  // We use another interval from tCurrentTimingIntervals
631  sUsedTimingIntervals++;
632 
633  // keep track of current timing offset
634  if (sUsedTimingIntervals >= sCurrentTimingIntervals) {
635  // we have used all intervals of current timing, switch to next timing value
636  sUsedTimingIntervals = 0;
637  sBiphaseDecodeRawbuffOffset++;
638  }
639 
640  TRACE_PRINTLN(tLevelOfCurrentInterval);
641 
642  return tLevelOfCurrentInterval;
643 }

685 bool IRrecv::decodeHash() {
686  long hash = FNV_BASIS_32;
687 
688 // Require at least 6 samples to prevent triggering on noise
689  if (decodedIRData.rawDataPtr->rawlen < 6) {
690  return false;
691  }
692 
693  for (unsigned int i = 1; (i + 2) < decodedIRData.rawDataPtr->rawlen; i++) {
694  uint8_t value = compare(decodedIRData.rawDataPtr->rawbuf[i], decodedIRData.rawDataPtr->rawbuf[i + 2]);
695  // Add value into the hash
696  hash = (hash * FNV_PRIME_32) ^ value;
697  }
698 
699  decodedIRData.decodedRawData = hash;
700  decodedIRData.numberOfBits = 32;
701  decodedIRData.protocol = UNKNOWN;
702 
703  return true;

1242 /**********************************************************************************************************************
1243  * Interrupt Service Routine - Called every 50 us
1244  *
1245  * Duration in ticks of 50 us of alternating SPACE, MARK are recorded in irparams.rawbuf array.
1246  * 'rawlen' counts the number of entries recorded so far.
1247  * First entry is the SPACE between transmissions.
1248  *
1249  * As soon as one SPACE entry gets longer than RECORD_GAP_TICKS, state switches to STOP (frame received). Timing of SPACE continues.
1250  * A call of resume() switches from STOP to IDLE.
1251  * As soon as first MARK arrives in IDLE, gap width is recorded and new logging starts.
1252  *
1253  * With digitalRead and Feedback LED
1254  * 15 pushs, 1 in, 1 eor before start of code = 2 us @16MHz + * 7.2 us computation time (6us idle time) + * pop + reti = 2.25 us @16MHz => 10.3 to 11.5 us @16MHz
1255  * With portInputRegister and mask and Feedback LED code commented
1256  * 9 pushs, 1 in, 1 eor before start of code = 1.25 us @16MHz + * 2.25 us computation time + * pop + reti = 1.5 us @16MHz => 5 us @16MHz
1257  * => Minimal CPU frequency is 4 MHz
1258  *
1259  **********************************************************************************************************************/
1260 //#define IR_MEASURE_TIMING
1261 //#define IR_TIMING_TEST_PIN 7 // do not forget to execute: "pinMode(IR_TIMING_TEST_PIN, OUTPUT);" if activated by line above
1262 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1263 #include "digitalWriteFast.h"
1264 #endif
1265 #if defined(TIMER_INTR_NAME)
1266 ISR (TIMER_INTR_NAME) // for ISR definitions
1267 #else
1268 ISR () // for functions definitions which are called by separate (board specific) ISR
1269 #endif
1270 {
1271 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1272  digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1273 #endif
1274 // 7 - 8.5 us for ISR body (without pushes and pops) for ATmega328 @16MHz
1275 
1276  TIMER_RESET_INTR_PENDING;// reset TickCounterForISR interrupt flag if required (currently only for Teensy and ATmega4809)
1277 
1278 // Read if IR Receiver -> SPACE [xmt LED off] or a MARK [xmt LED on]
1279 #if defined(__AVR__)
1280  uint8_t tIRInputLevel = *irparams.IRReceivePinPortInputRegister & irparams.IRReceivePinMask;
1281 #else
1282  uint8_t tIRInputLevel = (uint8_t) digitalRead(irparams.IRReceivePin);
1283 #endif
1284 
1285  /*
1286  * Increase TickCounter and clip it at maximum 0xFFFF / 3.2 seconds at 50 us ticks
1287  */
1288  if (irparams.TickCounterForISR < 0xFFFF) {
1289  irparams.TickCounterForISR++; // One more 50uS tick
1290  }
1291 
1292  /*
1293  * Due to a ESP32 compiler bug https://github.com/espressif/esp-idf/issues/1552 no switch statements are possible for ESP32
1294  * So we change the code to if / else if
1295  */
1296 // switch (irparams.StateForISR) {
1297 //......................................................................
1298  if (irparams.StateForISR == IR_REC_STATE_IDLE) { // In the middle of a gap or just resumed (and maybe in the middle of a transmission
1299  if (tIRInputLevel == INPUT_MARK) {
1300  // check if we did not start in the middle of a transmission by checking the minimum length of leading space
1301  if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
1302  // Gap just ended; Record gap duration + start recording transmission
1303  // Initialize all state machine variables
1304 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1305 // digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1306 #endif
1307  irparams.OverflowFlag = false;
1308  irparams.rawbuf[0] = irparams.TickCounterForISR;
1309  irparams.rawlen = 1;
1310  irparams.StateForISR = IR_REC_STATE_MARK;
1311  } // otherwise stay in idle state
1312  irparams.TickCounterForISR = 0;// reset counter in both cases
1313  }
1314 
1315  } else if (irparams.StateForISR == IR_REC_STATE_MARK) { // Timing Mark
1316  if (tIRInputLevel != INPUT_MARK) { // Mark ended; Record time
1317 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1318 // digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1319 #endif
1320  irparams.rawbuf[irparams.rawlen++] = irparams.TickCounterForISR;
1321  irparams.StateForISR = IR_REC_STATE_SPACE;
1322  irparams.TickCounterForISR = 0;
1323  }
1324 
1325  } else if (irparams.StateForISR == IR_REC_STATE_SPACE) { // Timing Space
1326  if (tIRInputLevel == INPUT_MARK) { // Space just ended; Record time
1327  if (irparams.rawlen >= RAW_BUFFER_LENGTH) {
1328  // Flag up a read OverflowFlag; Stop the State Machine
1329  irparams.OverflowFlag = true;
1330  irparams.StateForISR = IR_REC_STATE_STOP;
1331  } else {
1332 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1333 // digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1334 #endif
1335  irparams.rawbuf[irparams.rawlen++] = irparams.TickCounterForISR;
1336  irparams.StateForISR = IR_REC_STATE_MARK;
1337  }
1338  irparams.TickCounterForISR = 0;
1339 
1340  } else if (irparams.TickCounterForISR > RECORD_GAP_TICKS) {
1341  /*
1342  * Current code is ready for processing!
1343  * We received a long space, which indicates gap between codes.
1344  * Switch to IR_REC_STATE_STOP
1345  * Don't reset TickCounterForISR; keep counting width of next leading space
1346  */
1347  irparams.StateForISR = IR_REC_STATE_STOP;
1348  }
1349  } else if (irparams.StateForISR == IR_REC_STATE_STOP) {
1350  /*
1351  * Complete command received
1352  * stay here until resume() is called, which switches state to IR_REC_STATE_IDLE
1353  */
1354 #if defined(IR_MEASURE_TIMING) && defined(IR_TIMING_TEST_PIN)
1355 // digitalWriteFast(IR_TIMING_TEST_PIN, HIGH); // 2 clock cycles
1356 #endif
1357  if (tIRInputLevel == INPUT_MARK) {
1358  // Reset gap TickCounterForISR, to prepare for detection if we are in the middle of a transmission after call of resume()
1359  irparams.TickCounterForISR = 0;
1360  }
1361  }
1362 
1363 #if !defined(DISABLE_LED_FEEDBACK_FOR_RECEIVE)
1364  if (FeedbackLEDControl.LedFeedbackEnabled) {
1365  setFeedbackLED(tIRInputLevel == INPUT_MARK);
1366  }
1367 #endif
1368 
1369 #ifdef IR_MEASURE_TIMING
1370  digitalWriteFast(IR_TIMING_TEST_PIN, LOW); // 2 clock cycles
1371 #endif
1372 }
